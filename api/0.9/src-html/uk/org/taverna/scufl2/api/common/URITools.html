032            private static final String MERGE_POSITION = "mergePosition";
033            private static final String TO = "to";
034            private static final String FROM = "from";
035            private static final String DATALINK = "datalink";
036            private static final URI DOT = URI.create(".");
037    
038            public URI relativePath(URI base, URI uri) {
039    
040                    URI root = base.resolve("/");
041                    if (!root.equals(uri.resolve("/"))) {
042                            // Different protocol/host/auth
043                            return uri;
044                    }
045                    base = base.normalize();
046                    uri = uri.normalize();
047                    if (base.resolve("#").equals(uri.resolve("#"))) {
048                            // Same path, easy
049                            return base.relativize(uri);
050                    }
051    
052                    if (base.isAbsolute()) {
053                            // Ignore hostname and protocol
054                            base = root.relativize(base).resolve(".");
055                            uri = root.relativize(uri);
056                    }
057                    // Pretend they start from /
058                    base = root.resolve(base).resolve(".");
059                    uri = root.resolve(uri);
060    
061                    URI candidate = base.relativize(uri);
062                    URI relation = DOT;
063                    while (candidate.getPath().startsWith("/")
064                                    && !(base.getPath().isEmpty() || base.getPath().equals("/"))) {
065                            base = base.resolve("../");
066                            relation = relation.resolve("../");
067                            candidate = base.relativize(uri);
068                    }
069                    // Add the ../.. again
070                    URI resolved = relation.resolve(candidate);
071                    return resolved;
072    
073            }
074    
075            public URI relativeUriForBean(WorkflowBean bean, WorkflowBean relativeToBean) {
076                    URI rootUri = uriForBean(relativeToBean);
077                    URI beanUri = uriForBean(bean);
078                    return relativePath(rootUri, beanUri);
079            }

081            public WorkflowBean resolveUri(URI uri, WorkflowBundle wfBundle) {
082    
083                    // Check if it's a workflow URI
084                    for (Workflow wf : wfBundle.getWorkflows()) {
085                            if (wf.getWorkflowIdentifier().equals(uri)) {
086                                    return wf;
087                            }
088                    }
089                    String rel = Workflow.WORKFLOW_ROOT.relativize(uri).toASCIIString();
090                    if (rel.matches("[0-9a-f-]+/")) {
091                            return null;
092                    }
093    
094                    // Naive, super-inefficient reverse-lookup - we could have even returned
095                    // early!
096                    final Map<URI, WorkflowBean> uriToBean = new HashMap<URI, WorkflowBean>();
097                    wfBundle.accept(new VisitorWithPath() {
098                            @Override
099                            public boolean visit() {
100                                    WorkflowBean node = getCurrentNode();
101                                    URI uri;
102                                    try {
103                                            uri = uriForBean(node);
104                                    } catch (IllegalStateException ex) {
105                                            return false;
106                                    }
107                                    WorkflowBean existing = uriToBean.put(uri, node);
108                                    if (existing != null) {
109                                            String msg = "Multiple nodes with same URI {0}: {1} {2}";
110                                            throw new IllegalStateException(MessageFormat.format(msg,
111                                                            uri, existing, node));
112                                    }
113                                    return !(node instanceof Configuration);
114                            }
115                    });
116                    if (!uri.isAbsolute()) {
117                            // Make absolute, but remove / first
118                            uri = URI.create("/").relativize(uri);
119                            uri = uriForBean(wfBundle).resolve(uri);
120                    }
121                    return uriToBean.get(uri);
122            }

124            public URI uriForBean(WorkflowBean bean) {
125                    if (bean == null) {
126                            throw new NullPointerException("Bean can't be null");
127                    }
128                    if (bean instanceof Root) {
129                            Root root = (Root) bean;
130                            if (root.getGlobalBaseURI() == null) {
131                                    if (root instanceof WorkflowBundle) {
132                                            root.setGlobalBaseURI(WorkflowBundle.generateIdentifier());
133                                    } else {
134                                            throw new IllegalArgumentException(
135                                                            "sameBaseAs is null for bean " + bean);
136                                    }
137                            }
138                            return root.getGlobalBaseURI();
139                    }
140                    if (bean instanceof Child) {
141                            Child child = (Child) bean;
142                            WorkflowBean parent = child.getParent();
143                            if (parent == null) {
144                                    throw new IllegalStateException("Bean does not have a parent: "
145                                                    + child);
146                            }
147                            URI parentUri = uriForBean(parent);
148    
149                            if (!parentUri.getPath().endsWith("/")) {
150                                    parentUri = parentUri.resolve(parentUri.getPath() + "/");
151                            }
152                            String relation;
153                            if (child instanceof InputPort) {
154                                    relation = "in/";
155                            } else if (child instanceof OutputPort) {
156                                    relation = "out/";
157                            } else if (child instanceof IterationStrategyStack) {
158                                    relation = "iterationstrategy/";
159                            } else {
160                                    // TODO: Get relation by container annotations
161                                    relation = child.getClass().getSimpleName() + "/";
162                                    // Stupid fallback
163                            }
164    
165                            URI relationURI = parentUri.resolve(relation.toLowerCase());
166                            if (parent instanceof List) {
167                                    int index = ((List) parent).indexOf(child);
168                                    return parentUri.resolve(index + "/");
169                            }
170                            if (bean instanceof Named) {
171                                    Named named = (Named) bean;
172                                    String name = validFilename(named.getName());
173                                    if (!(bean instanceof Port)) {
174                                            name = name + "/";
175                                    }
176                                    return relationURI.resolve(name);
177                            } else if (bean instanceof DataLink) {
178    
179                                    DataLink dataLink = (DataLink) bean;
180                                    Workflow wf = dataLink.getParent();
181                                    URI wfUri = uriForBean(wf);
182                                    URI receivesFrom = relativePath(wfUri,
183                                                    uriForBean(dataLink.getReceivesFrom()));
184                                    URI sendsTo = relativePath(wfUri,
185                                                    uriForBean(dataLink.getSendsTo()));
186                                    String dataLinkUri = MessageFormat.format(
187                                                    "{0}?{1}={2}&{3}={4}", DATALINK, FROM, receivesFrom,
188                                                    TO, sendsTo);
189                                    if (dataLink.getMergePosition() != null) {
190                                            dataLinkUri += MessageFormat.format("&{0}={1}",
191                                                            MERGE_POSITION, dataLink.getMergePosition());
192                                    }
193                                    return wfUri.resolve(dataLinkUri);
194                            } else if (bean instanceof BlockingControlLink) {
195                                    BlockingControlLink runAfterCondition = (BlockingControlLink) bean;
196                                    Workflow wf = runAfterCondition.getParent();
197                                    URI wfUri = uriForBean(wf);
198    
199                                    URI start = relativePath(wfUri,
200                                                    uriForBean(runAfterCondition.getBlock()));
201                                    URI after = relativePath(wfUri,
202                                                    uriForBean(runAfterCondition.getUntilFinished()));
203                                    String conditionUri = MessageFormat.format(
204                                                    "{0}?{1}={2}&{3}={4}", "control", "block", start,
205                                                    "untilFinished", after);
206                                    return wfUri.resolve(conditionUri);
207                            } else if (bean instanceof DispatchStack) {
208                                    return relationURI;
209                            } else if (bean instanceof DispatchStackLayer) {
210                                    DispatchStackLayer dispatchStackLayer = (DispatchStackLayer) bean;
211    
212                            } else if (bean instanceof IterationStrategyStack) {
213                                    return relationURI;
214                            } else if (bean instanceof IterationStrategyNode) {
215                                    IterationStrategyNode iterationStrategyNode = (IterationStrategyNode) bean;
216                                    parent = iterationStrategyNode.getParent();
217                                    List<IterationStrategyNode> parentList = (List<IterationStrategyNode>) parent;
218                                    int index = parentList.indexOf(iterationStrategyNode);
219                                    return parentUri.resolve(index + "/");
220                            } else if (bean instanceof ProcessorPortBinding) {
221                                    // Named after the processor port, extract in/blah part.
222                                    ProcessorPortBinding<?, ?> processorPortBinding = (ProcessorPortBinding<?, ?>) bean;
223                                    ProcessorPort procPort = processorPortBinding
224                                                    .getBoundProcessorPort();
225                                    if (procPort == null) {
226                                            throw new IllegalStateException(
227                                                            "ProcessorPortBinding has no bound processor port: "
228                                                                            + bean);
229                                    }
230                                    URI procPortUri = relativeUriForBean(procPort,
231                                                    processorPortBinding.getParent().getBoundProcessor());
232                                    return parentUri.resolve(procPortUri);
233                            } else {
234                                    throw new IllegalStateException(
235                                                    "Can't create URIs for non-named child: " + bean);
236                            }
237    
238                    }
239                    if (bean instanceof PropertyResource) {
240                            PropertyResource propertyResource = (PropertyResource) bean;
241                            URI resourceURI = propertyResource.getResourceURI();
242                            if (resourceURI != null) {
243                                    return resourceURI;
244                            }
245                            throw new IllegalStateException(
246                                            "PropertyResource does not have a resourceURI");
247                    }
248                    throw new IllegalArgumentException("Unsupported type "
249                                    + bean.getClass() + " for bean " + bean);
250            }

252            public String validFilename(String name) {
253                    if (name == null) {
254                            throw new NullPointerException();
255                    }
256                    // Make a relative URI
257                    URI uri;
258                    try {
259                            uri = new URI(null, null, name, null, null);
260                    } catch (URISyntaxException e) {
261                            throw new IllegalArgumentException("Invalid name " + name);
262                    }
263                    String ascii = uri.toASCIIString();
264                    // And escape / and \
265                    String escaped = ascii.replace("/", "%2f");
266                    // escaped = escaped.replace("\\", "%5c");
267                    escaped = escaped.replace(":", "%3a");
268                    escaped = escaped.replace("=", "%3d");
269                    return escaped;
270            }
