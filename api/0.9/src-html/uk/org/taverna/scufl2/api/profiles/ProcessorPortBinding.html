001    package uk.org.taverna.scufl2.api.profiles;
002    
003    import uk.org.taverna.scufl2.api.common.Child;
004    import uk.org.taverna.scufl2.api.port.ActivityPort;
005    import uk.org.taverna.scufl2.api.port.ProcessorPort;
006    
007    /**
008     * The binding between an <code>ActivityPort</code> and a <code>ProcessorPort</code>.
009     * 
010     * @param <A>
011     *            the <code>ActivityPort</code>
012     * @param <P>
013     *            the <code>ProcessorPort</code>
014     */
015    public interface ProcessorPortBinding<A extends ActivityPort, P extends ProcessorPort> extends
016    Child<ProcessorBinding> {
017    
018            /**
019             * Returns the bound <code>ActivityPort</code>
020             * 
021             * @return the bound <code>ActivityPort</code>
022             */
023            A getBoundActivityPort();
024    
025            /**
026             * Returns the bound <code>ProcessorPort</code>
027             * 
028             * @return the bound <code>ProcessorPort</code>
029             */
030            P getBoundProcessorPort();
031    
032            /**
033             * Sets the bound <code>ProcessorPort</code>
034             * 
035             * @param boundActivityPort
036             *            the bound <code>ProcessorPort</code>
037             */
038            void setBoundActivityPort(A boundActivityPort);
039    
040            /**
041             * Sets the bound <code>ProcessorPort</code>
042             * 
043             * @param boundProcessorPort
044             *            the bound <code>ProcessorPort</code>
045             */
046            void setBoundProcessorPort(P boundProcessorPort);
047    
048    }
