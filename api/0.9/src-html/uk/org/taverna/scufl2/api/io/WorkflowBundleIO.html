031            protected List<WorkflowBundleReader> discoverReaders() {
032                    synchronized (this) {
033    
034                            if (readersLoader == null) {
035                                    // FIXME: This uses Thread.currentThread.getContextClassLoader()
036                                    // - depending on who gets synchronized-block first this can
037                                    // vary - but as it's still local per instance of
038                                    // WorkflowBundleIO it should be OK for now..
039                                    readersLoader = ServiceLoader.load(WorkflowBundleReader.class);
040                            }
041                    }
042    
043                    List<WorkflowBundleReader> allReaders = new ArrayList<WorkflowBundleReader>();
044                    for (WorkflowBundleReader reader : readersLoader) {
045                            allReaders.add(reader);
046                    }
047                    return allReaders;
048            }
049    
050            protected List<WorkflowBundleWriter> discoverWriters() {
051                    synchronized (this) {
052                            if (writersLoader == null) {
053                                    // FIXME: This uses Thread.currentThread.getContextClassLoader()
054                                    // - depending on who gets synchronized-block first this can
055                                    // vary - but as it's still local per instance of
056                                    // WorkflowBundleIO it should be OK for now..
057                                    writersLoader = ServiceLoader.load(WorkflowBundleWriter.class);
058                            }
059                    }
060                    List<WorkflowBundleWriter> allWriters = new ArrayList<WorkflowBundleWriter>();
061                    for (WorkflowBundleWriter writer : writersLoader) {
062                            allWriters.add(writer);
063                    }
064                    return allWriters;
065            }

067            /**
068             * Returns a <code>WorkflowBundleReader</code> for the specified media type.
069             * 
070             * If there is more than one <code>WorkflowBundleReader</code> for the specified media type the
071             * first reader discovered is returned. Subsequent calls to this method may return a different
072             * reader.
073             * 
074             * If there is no <code>WorkflowBundleReader</code> for the specified media type
075             * <code>null</code> is returned.
076             * 
077             * @param mediaType
078             *            the media type of the <code>WorkflowBundleReader</code>
079             * @return a <code>WorkflowBundleReader</code> for the specified media type
080             */
081            public WorkflowBundleReader getReaderForMediaType(String mediaType) {
082                    for (WorkflowBundleReader reader : getReaders()) {
083                            if (reader.getMediaTypes().contains(mediaType)) {
084                                    return reader;
085                            }
086                    }
087                    return null;
088            }
089    
090            /**
091             * Returns all the available <code>WorkflowBundleReader</code>s.
092             * 
093             * @return all the available <code>WorkflowBundleReader</code>s
094             */
095            public List<WorkflowBundleReader> getReaders() {
096                    if (readers == null) {
097                            return discoverReaders();
098                    }
099                    return readers;
100            }

137            /**
138             * Attempt to guess the media type for a stream or file that starts with
139             * these bytes. 
140             * <p>
141             * All registered {@link #getReaders()} are consulted. 
142             * <p>
143             * Return <code>null</code> if ambiguous (more than one possibility) or
144             * unknown.
145             * 
146             * @param firstBytes
147             *            The initial bytes, at least 512 bytes long unless the resource
148             *            is smaller.
149             * @return The recognised media type, or <code>null</code> if the bytes were
150             *         ambiguous or unknown.
151             */
152            public String guessMediaTypeForSignature(byte[] firstBytes) {
153                    Set<String> mediaTypes = new HashSet<String>();
154                    for (WorkflowBundleReader reader : getReaders()) {
155                            String guess = reader.guessMediaTypeForSignature(firstBytes);
156                            if (guess != null) {
157                                    mediaTypes.add(guess);
158                            }
159                    }
160                    if (mediaTypes.isEmpty()) {
161                            return null;
162                    }
163                    if (mediaTypes.size() > 1) {
164                            // log.warn("Multiple media types found: " + mediaTypes)
165                            return null;
166                    }
167                    return mediaTypes.iterator().next();
168            }

171            /**
172             * Reads a file containing a workflow bundle in the specified media type and returns a
173             * <code>WorkflowBundle</code>.
174             * 
175             * @param bundleFile
176             *            the file containing the workflow bundle
177             * @param mediaType
178             *            the media type of the workflow bundle. A <code>WorkflowBundleReader</code> must
179             *            exist for this media type.
180             *            If <code>null</code>, the media type will be guessed as with {@link #guessMediaTypeForSignature(byte[])}.
181             * @return the <code>WorkflowBundle</code> read from the file
182             * @throws ReaderException
183             *             if there is an error parsing the workflow bundle
184             * @throws IOException
185             *             if there is an error reading the file
186             * @throws IllegalArgumentException
187             *             if a <code>WorkflowBundleReader</code> cannot be found for the media type
188             */
189            public WorkflowBundle readBundle(File bundleFile, String mediaType) throws ReaderException,
190            IOException {
191                    if (mediaType == null) {
192                            byte[] firstBytes = new byte[1024];
193                            new FileInputStream(bundleFile).read(firstBytes);
194                            mediaType = guessMediaTypeForSignature(firstBytes);
195                    }
196                    WorkflowBundleReader reader = getReaderForMediaType(mediaType);
197                    if (reader == null) {
198                            if (mediaType == null) {
199                                    throw new IllegalArgumentException("Could not guess media type for " + bundleFile);
200                            }
201                            throw new IllegalArgumentException("Could not find reader for media type " + mediaType);
202                    }
203                    return reader.readBundle(bundleFile, mediaType);
204            }
205    
206            /**
207             * Reads a stream containing a workflow bundle in the specified media type and returns a
208             * <code>WorkflowBundle</code>.
209             * 
210             * @param inputStream
211             *            the stream containing the workflow bundle
212             * @param mediaType
213             *            the media type of the workflow bundle. A <code>WorkflowBundleReader</code> must
214             *            exist for this media type.
215             *            If <code>null</code>, the media type will be guessed as with {@link #guessMediaTypeForSignature(byte[])}.
216             * @return the <code>WorkflowBundle</code> read from the stream
217             * @throws ReaderException
218             *             if there is an error parsing the workflow bundle
219             * @throws IOException
220             *             if there is an error reading from the stream
221             * @throws IllegalArgumentException
222             *             if a <code>WorkflowBundleReader</code> cannot be found for the media type
223             */
224            public WorkflowBundle readBundle(InputStream inputStream, String mediaType)
225            throws ReaderException, IOException {
226                    
227                    if (mediaType == null) {
228                            byte[] firstBytes = new byte[1024];
229                            inputStream = new BufferedInputStream(inputStream);
230                            try { 
231                                    inputStream.mark(firstBytes.length*2);                  
232                                    inputStream.read(firstBytes);
233                                    mediaType = guessMediaTypeForSignature(firstBytes);
234                            } finally {
235                                    inputStream.reset();
236                                    // Important = so readBundle can start from the beginning
237                            }
238                    }
239                    WorkflowBundleReader reader = getReaderForMediaType(mediaType);
240                    if (reader == null) {
241                            if (mediaType == null) {
242                                    throw new IllegalArgumentException("Could not guess media type for input stream");
243                            }
244                            throw new IllegalArgumentException("Could not find reader for media type " + mediaType);
245                    }
246                    return reader.readBundle(inputStream, mediaType);
247            }

257             *            exist for this media type
258             *            If <code>null</code>, the media type will 
259             *            be guessed as with {@link #guessMediaTypeForSignature(byte[])}.

263             * @throws IOException
264             *             if there is an error reading from the stream
265             * @throws IllegalArgumentException
266             *             if a <code>WorkflowBundleReader</code> cannot be found for the media type
267             */
268            public WorkflowBundle readBundle(URL url, String mediaType) throws ReaderException, IOException {
269                    // TODO: Pass URL to reader
270                    // TODO: Use known media type for Accept header
271                    // TODO: Extract Content-Type from result
272                    return readBundle(url.openStream(), mediaType);

300            /**
301             * Writes a <code>WorkflowBundle</code> to a file with specified media type.
302             * 
303             * @param wfBundle
304             *            the workflow bundle to write
305             * @param destination
306             *            the file to write the workflow bundle to
307             * @param mediaType
308             *            the media type to write workflow bundle in. A <code>WorkflowBundleWriter</code>
309             *            must exist for this media type
310             * @throws WriterException
311             *             if there is an error writing the workflow bundle
312             * @throws IOException
313             *             if there is an error writing the file
314             * @throws IllegalArgumentException
315             *             if a <code>WorkflowBundleWriter</code> cannot be found for the media type
316             */
317            public void writeBundle(WorkflowBundle wfBundle, File destination, String mediaType)
318            throws WriterException, IOException {
319                    WorkflowBundleWriter writer = getWriterForMediaType(mediaType);
320                    if (writer == null) {
321                            throw new IllegalArgumentException("Could not find writer for media type " + mediaType);
322                    }
323                    writer.writeBundle(wfBundle, destination, mediaType);
324            }

337             *             if there is an error writing the workflow bundle

340             * @throws IllegalArgumentException
341             *             if a <code>WorkflowBundleWriter</code> cannot be found for the media type
