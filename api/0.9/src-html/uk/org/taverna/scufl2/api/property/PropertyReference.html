001    package uk.org.taverna.scufl2.api.property;
002    
003    import java.net.URI;
004    
005    import uk.org.taverna.scufl2.api.common.Visitor;
006    
007    /**
008     * A {@link PropertyObject} representing a reference to a value.
009     */
010    public class PropertyReference implements PropertyObject {
011    
012            private URI resourceURI;
013    
014            public PropertyReference() {
015                    super();
016            }
017    
018            public PropertyReference(URI resourceURI) {
019                    this.resourceURI = resourceURI;
020            }
021    
022            @Override
023            public boolean accept(Visitor visitor) {
024                    return visitor.visit(this);
025            }
026    
027            public final URI getResourceURI() {
028                    return resourceURI;
029            }
030    
031            public final void setResourceURI(URI resourceURI) {
032                    this.resourceURI = resourceURI;
033            }
034    
035            @Override
036            public String toString() {
037                    return "PropertyReference [getResourceURI()=" + getResourceURI() + "]";
038            }
039    
040    }
