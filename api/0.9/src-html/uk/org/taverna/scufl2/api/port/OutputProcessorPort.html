001    package uk.org.taverna.scufl2.api.port;
002    
003    import uk.org.taverna.scufl2.api.core.Processor;
004    
005    
006    /**
007     * An <code>OutputProcessorPort</code> is a <Port> that outputs data from a {@link Processor}.
008     */
009    public class OutputProcessorPort extends AbstractGranularDepthPort implements
010    SenderPort, ProcessorPort, OutputPort, GranularDepthPort {
011    
012    
013    
014            private Processor parent;
015    
016            /**
017             * Constructs an <code>OutputProcessorPort</code> with a random UUID as the name.
018             */
019            public OutputProcessorPort() {
020                    super();
021            }
022    
023            /**
024             * Constructs an <code>OutputProcessorPort</code> for the specified <code>Processor</code> with the
025             * specified name.
026             * <p>
027             * The <code>OutputPort</code> is added to the <code>Processor</code> (if the
028             * <code>Processor</code> is not <code>null</code>).
029             * 
030             * @param parent
031             *            the <code>Processor</code> to add this <code>Port</code> to. Can be
032             *            <code>null</code>
033             * @param name
034             *            the name of the <code>Port</code>. <strong>Must not</strong> be <code>null</code>
035             *            or an empty String.
036             */
037            public OutputProcessorPort(Processor parent, String name) {
038                    super(name);
039                    setParent(parent);
040            }
041    
042            /* (non-Javadoc)
043             * @see uk.org.taverna.scufl2.api.common.Child#getParent()
044             */
045            @Override
046            public Processor getParent() {
047                    return parent;
048            }
049    
050            /* (non-Javadoc)
051             * @see uk.org.taverna.scufl2.api.common.Child#setParent(uk.org.taverna.scufl2.api.common.WorkflowBean)
052             */
053            @Override
054            public void setParent(Processor parent) {
055                    if (this.parent != null && this.parent != parent) {
056                            this.parent.getOutputPorts().remove(this);
057                    }
058                    this.parent = parent;
059                    if (parent != null) {
060                            parent.getOutputPorts().add(this);
061                    }
062            }
063    
064            @Override
065            public String toString() {
066                    return getParent().getName() + "." + getName();
067            }
068    }
