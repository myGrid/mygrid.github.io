001    package uk.org.taverna.scufl2.api.profiles;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    import java.util.Set;
006    
007    import uk.org.taverna.scufl2.api.activity.Activity;
008    import uk.org.taverna.scufl2.api.common.AbstractNamedChild;
009    import uk.org.taverna.scufl2.api.common.Child;
010    import uk.org.taverna.scufl2.api.common.NamedSet;
011    import uk.org.taverna.scufl2.api.common.Visitor;
012    import uk.org.taverna.scufl2.api.common.WorkflowBean;
013    import uk.org.taverna.scufl2.api.configurations.Configuration;
014    import uk.org.taverna.scufl2.api.container.WorkflowBundle;
015    
016    /**
017     * A <code>Profile</code> specifies a set of compatible {@link ProcessorBinding}s.
018     * <p>
019     * For example, one <code>Profile</code> could contain ways of enacting a set of
020     * {@link uk.org.taverna.scufl2.api.core.Processor Processor}s on a grid whilst another contained
021     * ways of enacting the <code>Processor</code>s on a laptop.
022     * 
023     * @author Alan R Williams
024     */
025    public class Profile extends AbstractNamedChild implements Child<WorkflowBundle> {
026    
027            private final NamedSet<ProcessorBinding> processorBindings = new NamedSet<ProcessorBinding>();
028    
029            private final NamedSet<Configuration> configurations = new NamedSet<Configuration>();
030    
031            private Integer profilePosition;
032    
033            private final NamedSet<Activity> activities = new NamedSet<Activity>();
034    
035            private WorkflowBundle parent;
036    
037            /**
038             * Constructs a <code>Profile</code> with a random UUID as the name.
039             */
040            public Profile() {
041                    super();
042            }
043    
044            /**
045             * Constructs a <code>Profile</code> with the specified name.
046             * 
047             * @param name
048             *            the name of the <code>Port</code>. <strong>Must not</strong> be <code>null</code>
049             *            or an empty String.
050             */
051            public Profile(String name) {
052                    super(name);
053            }
054    
055            @Override
056            public boolean accept(Visitor visitor) {
057                    if (visitor.visitEnter(this)) {
058                            List<Iterable<? extends WorkflowBean>> children = new ArrayList<Iterable<? extends WorkflowBean>>();
059                            children.add(getActivities());
060                            children.add(getProcessorBindings());
061                            children.add(getConfigurations());
062                            outer: for (Iterable<? extends WorkflowBean> it : children) {
063                                    for (WorkflowBean bean : it) {
064                                            if (!bean.accept(visitor)) {
065                                                    break outer;
066                                            }
067                                    }
068                            }
069                    }
070                    return visitor.visitLeave(this);
071            }
072    
073            /**
074             * Returns the <code>Activity</code>s that this <code>Profile</code> contains.
075             * <p>
076             * The <code>Activity</code>s may be bound to <code>Processor</code>s in the
077             * <code>ProcessorBinding</code>s.
078             * 
079             * @return the <code>Activity</code>s that this <code>Profile</code> contains
080             */
081            public NamedSet<Activity> getActivities() {
082                    return activities;
083            }
084    
085            /**
086             * Returns the <code>Configuration</code>s that this <code>Profile</code> contains.
087             * 
088             * @return the <code>Configuration</code>s that this <code>Profile</code> contains
089             */
090            public NamedSet<Configuration> getConfigurations() {
091                    return configurations;
092            }
093    
094            @Override
095            public WorkflowBundle getParent() {
096                    return parent;
097            }
098    
099            /**
100             * Return the set of bindings for individual <code>Processor</code>s.
101             * 
102             * @return the set of bindings for individual <code>Processor</code>s
103             */
104            public NamedSet<ProcessorBinding> getProcessorBindings() {
105                    return processorBindings;
106            }
107    
108            /**
109             * Return the suggested position of this profile within the {@link WorkflowBundle}.
110             * <p>
111             * If ordering profiles (for instance for displaying them to the user), they 
112             * might be sorted by increasing profilePosition. If two profiles
113             * have the same position, their internal order is undetermined. Profiles with 
114             * profile position <code>null</code> should be sorted last in such a list. 
115             * 
116             * @return the position of this profile within the <code>WorkflowBundle</code>
117             */
118            public final Integer getProfilePosition() {
119                    return profilePosition;
120            }
121    
122            /**
123             * Sets the <code>Activity</code>s that this <code>Profile</code> contains.
124             * 
125             * @param activities the <code>Activity</code>s that this <code>Profile</code> contains
126             */
127            public void setActivities(Set<Activity> activities) {
128                    this.activities.clear();
129                    this.activities.addAll(activities);
130            }
131    
132            /**
133             * Sets the <code>Configuration</code>s that this <code>Profile</code> contains.
134             * 
135             * @param configurations the <code>Configuration</code>s that this <code>Profile</code> contains
136             */
137            public void setConfigurations(Set<Configuration> configurations) {
138                    this.configurations.clear();
139                    this.configurations.addAll(configurations);
140            }
141    
142            @Override
143            public void setParent(WorkflowBundle parent) {
144                    if (this.parent != null && this.parent != parent) {
145                            this.parent.getProfiles().remove(this);
146                    }
147                    this.parent = parent;
148                    if (parent != null) {
149                            parent.getProfiles().add(this);
150                    }
151    
152            }
153    
154            /**
155             * Sets the bindings for individual <code>Processor</code>s.
156             * 
157             * @param processorBindings the bindings for individual <code>Processor</code>s
158             */
159            public void setProcessorBindings(Set<ProcessorBinding> processorBindings) {
160                    this.processorBindings.clear();
161                    this.processorBindings.addAll(processorBindings);
162            }
163    
164            /**
165             * Sets the position of this profile within the {@link WorkflowBundle}.
166             * <p>
167             * When ordering profiles, they can be sorted by decreasing profilePosition. If two profiles
168             * have the same position, their internal order is undetermined.
169             * 
170             * 
171             * @param profilePosition
172             *            the position of this profile within the <code>WorkflowBundle</code>
173             */
174            public final void setProfilePosition(Integer profilePosition) {
175                    this.profilePosition = profilePosition;
176            }
177    
178    }
