001    package uk.org.taverna.scufl2.api.common;
002    
003    import java.net.URI;
004    import java.util.UUID;
005    
006    /**
007     * Abstract implementation of a {@link Named} {@link WorkflowBean}.
008     * 
009     * @author Alan R Williams
010     * @author Stian Soiland-Reyes
011     */
012    public abstract class AbstractNamed implements Named {
013    
014            private String name;
015    
016            /**
017             * Constructs a {@link Named} {@link WorkflowBean} with a random UUID as the name.
018             */
019            public AbstractNamed() {
020                    setName(UUID.randomUUID().toString());
021            }
022    
023            /**
024             * Constructs a {@link Named} {@link WorkflowBean} with the specified name.
025             * 
026             * @param name
027             *            the name of the <code>Named</code> <code>WorkflowBean</code>. <strong>Must not</strong> be <code>null</code>
028             *            or an empty String.
029             */
030            public AbstractNamed(String name) {
031                    setName(name);
032            }
033    
034            @Override
035            public int compareTo(Object o) {
036                    if (!(o instanceof AbstractNamed)) {
037                            // Other comparables go first
038                            return 1;
039                    }
040                    AbstractNamed other = (AbstractNamed) o;
041                    if (other == this) {
042                            return 0;
043                    }
044                    if (this instanceof Child) {
045                            if (!(other instanceof Child)) {
046                                    // He's orphan, he's sorted first
047                                    return 1;
048                            }
049                            WorkflowBean parent = ((Child<?>) this).getParent();
050                            WorkflowBean otherParent = ((Child<?>) other).getParent();
051                            if (parent instanceof Comparable && otherParent instanceof Comparable) {
052                                    int comparedParents = ((Comparable) parent).compareTo(otherParent);
053                                    if (comparedParents != 0) {
054                                            return comparedParents;
055                                    }
056                            }
057                    } else {
058                            if (other instanceof Child) {
059                                    // We're orphan, we're first
060                                    return -1;
061                            }
062                    }
063                    if (getClass() != other.getClass()) {
064                            int classCompare = getClass().getCanonicalName().compareTo(
065                                            other.getClass().getCanonicalName());
066                            if (classCompare != 0) {
067                                    // Allow having say InputPorts and OutputPorts in the same sorted list
068                                    return classCompare;
069                            }
070                    }
071                    // We're the same class, let's compare the names
072                    return getName().compareTo(other.getName());
073            }
074    
075            @Override
076            public boolean equals(Object obj) {
077                    if (this == obj) {
078                            return true;
079                    }
080                    if (obj == null) {
081                            return false;
082                    }
083                    if (getClass() != obj.getClass()) {
084                            return false;
085                    }
086                    AbstractNamed other = (AbstractNamed) obj;
087                    if (!getName().equals(other.getName())) {
088                            return false;
089                    }
090                    if (this instanceof Child) {
091                            WorkflowBean parent = ((Child<?>) this).getParent();
092                            WorkflowBean otherParent = ((Child<?>) other).getParent();
093                            if (parent != null) {
094                                    return parent.equals(otherParent);
095                            }
096                            if (parent == null && otherParent != null) {
097                                    return false;
098                            }
099                    }
100                    if (this instanceof Typed) {
101                            URI myId = ((Typed) this).getConfigurableType();
102                            URI otherId = ((Typed) obj).getConfigurableType();
103                            if (myId != null) {
104                                    return myId.equals(otherId);
105                            }
106                            if (myId == null && otherId != null) {
107                                    return false;
108                            }
109                    }
110                    return true;
111            }
112    
113            /*
114             * (non-Javadoc)AbstractNamed
115             * 
116             * @see uk.org.taverna.scufl2.api.common.Named#getName()
117             */
118            public String getName() {
119                    return name;
120            }
121    
122            @Override
123            public int hashCode() {
124                    final int prime = 31;
125                    int result = 1;
126                    result = prime * result + (name == null ? 0 : name.hashCode());
127    
128                    if (this instanceof Child) {
129                            WorkflowBean parent = ((Child) this).getParent();
130                            if (parent != null) {
131                                    result = prime * result + parent.hashCode();
132                            }
133                    }
134                    return result;
135            }
136    
137            /*
138             * (non-Javadoc)
139             * 
140             * @see uk.org.taverna.scufl2.api.common.Named#setName(java.lang.String)
141             */
142            @SuppressWarnings("unchecked")
143            public void setName(String name) {
144                    if (name == null) {
145                            throw new NullPointerException("Name can't be null");
146                    }
147                    if (name.length() == 0) {
148                            throw new IllegalArgumentException("Name can't be empty");
149                    }
150    
151                    if (this instanceof Child) {
152                            Child child = (Child) this;
153                            WorkflowBean parent = child.getParent();
154                            if (parent != null) {
155                                    child.setParent(null);
156                                    this.name = name;
157                                    // Might overwrite other Named object with same name
158                                    child.setParent(parent);
159                            }
160                    }
161                    this.name = name;
162            }
163    
164            @Override
165            public String toString() {
166                    return getClass().getSimpleName() + " \"" + getName() + '"';
167            }
168    
169    }
