017    /**
018     * A {@link PropertyObject} representing a compound property that contains {@link PropertyObject}s.
019     */
020    public class PropertyResource extends PropertyReference implements
021    PropertyObject {
022    
023            public static class PropertyComparator implements
024            Comparator<PropertyObject> {
025                    private static final List<Class<?>> CLASSORDERING = Arrays
026                    .<Class<?>> asList(PropertyLiteral.class, PropertyList.class,
027                                    PropertyReference.class, PropertyResource.class);
028    
029                    @Override
030                    public int compare(PropertyObject o1, PropertyObject o2) {
031                            if (o1 == o2) {
032                                    return 0;
033                            }
034                            if (o1.getClass() != o2.getClass()) {
035                                    return CLASSORDERING.indexOf(o1.getClass())
036                                    - CLASSORDERING.indexOf(o2.getClass());
037                            }
038                            if (o1 instanceof PropertyLiteral) {
039                                    return compareLiteral((PropertyLiteral) o1,
040                                                    (PropertyLiteral) o2);
041                            }
042                            if (o1 instanceof PropertyResource) {
043                                    return compareResource((PropertyResource) o1,
044                                                    (PropertyResource) o2);
045                            }
046                            if (o1 instanceof PropertyReference) {
047                                    // Remember to check PropertyReference *after* PropertyResource
048                                    // which is a subclass of PropertyReference
049                                    return compareReference((PropertyReference) o1,
050                                                    (PropertyReference) o2);
051                            }
052                            if (o1 instanceof PropertyList) {
053                                    return compareList((PropertyList) o1, (PropertyList) o2);
054                            }
055                            throw new IllegalArgumentException(
056                                            "Comparison of unsupported subclass " + o1.getClass());
057                    }
058    
059                    private int compareIterators(Iterator<PropertyObject> it1,
060                                    Iterator<PropertyObject> it2) {
061                            while (it1.hasNext() && it2.hasNext()) {
062                                    int v = compare(it1.next(), it2.next());
063                                    if (v != 0) {
064                                            return v;
065                                    }
066                            }
067                            if (it1.hasNext() && !it2.hasNext()) {
068                                    return 1; // it2 is 'smaller' than us
069                            } else if (!it1.hasNext() && it2.hasNext()) {
070                                    return -1;
071                            }
072                            return 0;
073    
074                    }
075    
076                    private int compareList(PropertyList o1, PropertyList o2) {
077                            return compareIterators(o1.iterator(), o2.iterator());
078                    }
079    
080                    private int compareLiteral(PropertyLiteral o1, PropertyLiteral o2) {
081                            int v = o1.getLiteralValue().compareTo(o2.getLiteralValue());
082                            if (v != 0) {
083                                    return v;
084                            }
085                            return compareWithNullCheck(o1.getLiteralType(),
086                                            o2.getLiteralType());
087                    }
088    
089                    private int compareMaps(Map<URI, SortedSet<PropertyObject>> map,
090                                    Map<URI, SortedSet<PropertyObject>> map2) {
091                            TreeSet<URI> keys = new TreeSet<URI>();
092                            keys.addAll(map.keySet());
093                            keys.addAll(map2.keySet());
094    
095                            for (URI uri : keys) {
096                                    Set<PropertyObject> set1 = map.get(uri);
097                                    Set<PropertyObject> set2 = map2.get(uri);
098                                    if (set1 == null && set2 == null) {
099                                            continue;
100                                    }
101                                    if (set1 == null && set2 != null) {
102                                            return 1; // set2 is there before us
103                                    }
104                                    if (set2 == null && set1 != null) {
105                                            return -1;
106                                    }
107    
108                                    Iterator<PropertyObject> it1 = set1.iterator();
109                                    Iterator<PropertyObject> it2 = set2.iterator();
110                                    int v = compareIterators(it1, it2);
111                                    if (v != 0) {
112                                            return v;
113                                    }
114                            }
115                            // all (or none) equal
116                            return 0;
117                    }
118    
119                    private int compareReference(PropertyReference o1, PropertyReference o2) {
120                            return compareWithNullCheck(o1.getResourceURI(),
121                                            o2.getResourceURI());
122                    }
123    
124                    private int compareResource(PropertyResource o1, PropertyResource o2) {
125                            int diff = compareWithNullCheck(o1.getResourceURI(),
126                                            o2.getResourceURI());
127                            if (diff != 0) {
128                                    return diff;
129                            }
130    
131                            diff = compareWithNullCheck(o1.getTypeURI(), o2.getTypeURI());
132                            if (diff != 0) {
133                                    return diff;
134                            }
135    
136                            return compareMaps(o1.getProperties(), o2.getProperties());
137                    }
138    
139                    private <T extends Comparable<T>> int compareWithNullCheck(T o1, T o2) {
140                            if (o1 == o2 || o1 == null && o2 == null) {
141                                    return 0;
142                            }
143                            if (o1 != null && o2 == null) {
144                                    // Null is 'more than' having a value
145                                    return -1;
146                            }
147                            if (o1 == null && o2 != null) {
148                                    return 1;
149                            }
150                            return o1.compareTo(o2);
151                    }
152    
153            }

155            /**
156             * A special {@link Child} used by {@link PropertyResource#accept(Visitor)}
157             * when visiting the map of {@link PropertyResource#getProperties()}
158             *
159             * @author Stian Soiland-Reyes
160             *
161             */
162            public class PropertyVisit implements Child<PropertyResource> {
163    
164                    private final URI predicateUri;
165    
166                    PropertyVisit(URI uri) {
167                            predicateUri = uri;
168                    }
169    
170                    @Override
171                    public boolean accept(Visitor visitor) {
172                            if (visitor.visitEnter(this)) {
173                                    for (PropertyObject po : getPropertiesForPredicate()) {
174                                            if (!po.accept(visitor)) {
175                                                    break;
176                                            }
177                                    }
178                            }
179                            return visitor.visitLeave(this);
180                    }
181    
182                    @Override
183                    public PropertyResource getParent() {
184                            return PropertyResource.this;
185                    }
186    
187                    public URI getPredicateUri() {
188                            return predicateUri;
189                    }
190    
191                    public Set<PropertyObject> getPropertiesForPredicate() {
192                            return getProperties().get(predicateUri);
193                    }
194    
195                    @Override
196                    public void setParent(PropertyResource parent) {
197                            throw new UnsupportedOperationException();
198                    }
199    
200                    @Override
201                    public String toString() {
202                            return "PropertyVisit " + getPredicateUri();
203                    }
204    
205            }
206    
207            private URI typeURI;
208    
209            private final Map<URI, SortedSet<PropertyObject>> properties = new LazyMap<URI, SortedSet<PropertyObject>>() {
210                    private static final long serialVersionUID = 1L;
211    
212                    @Override
213                    public SortedSet<PropertyObject> getDefault(URI key) {
214                            return new TreeSet<PropertyObject>(new PropertyComparator());
215                    }
216            };
217    
218            public PropertyResource() {
219            }
220    
221            @Override
222            public boolean accept(Visitor visitor) {
223                    if (visitor.visitEnter(this)) {
224                            Set<URI> uris = getProperties().keySet();
225                            for (URI uri : uris) {
226                                    if (!new PropertyVisit(uri).accept(visitor)) {
227                                            break;
228                                    }
229                            }
230                    }
231                    return visitor.visitLeave(this);
232            }

234            /**
235             * Adds a property
236             * 
237             * @param predicate
238             * @param object
239             */
240            public void addProperty(URI predicate, PropertyObject object) {
241                    getProperties().get(predicate).add(object);
242            }
243    
244            /**
245             * 
246             * 
247             * @param predicate
248             * @param typeURI
249             * @return
250             */
251            public PropertyResource addPropertyAsNewResource(URI predicate, URI typeURI) {
252                    PropertyResource resource = new PropertyResource();
253                    resource.setTypeURI(typeURI);
254                    addProperty(predicate, resource);
255                    return resource;
256            }

258            /**
259             * 
260             * 
261             * @param predicate
262             * @param value
263             */
264            public void addPropertyAsString(URI predicate, String value) {
265                    addProperty(predicate, new PropertyLiteral(value));
266            }

269             * 
270             * 
271             * @param predicate
272             * @param resourceURI
273             */
274            public void addPropertyReference(URI predicate, URI resourceURI) {
275                    addProperty(predicate, new PropertyReference(resourceURI));
276            }

278            public final Map<URI, SortedSet<PropertyObject>> getProperties() {
279                    return properties;
280            }

283            public SortedSet<PropertyLiteral> getPropertiesAsLiterals(URI predicate)
284            throws UnexpectedPropertyException {
285                    return getPropertiesOfType(predicate, PropertyLiteral.class);
286            }
287    
288            public SortedSet<PropertyReference> getPropertiesAsReferences(URI predicate)
289            throws UnexpectedPropertyException {
290                    return getPropertiesOfType(predicate, PropertyReference.class);
291            }
292    
293            public SortedSet<PropertyResource> getPropertiesAsResources(URI predicate)
294            throws UnexpectedPropertyException {
295                    return getPropertiesOfType(predicate, PropertyResource.class);
296            }
297    
298            public SortedSet<URI> getPropertiesAsResourceURIs(URI predicate)
299            throws UnexpectedPropertyException {
300                    SortedSet<URI> uris = new TreeSet<URI>();
301                    for (PropertyReference resource : getPropertiesAsReferences(predicate)) {
302                            URI uri = resource.getResourceURI();
303                            if (uri == null) {
304                                    throw new UnexpectedPropertyException(
305                                                    "Resource property without URI for " + predicate
306                                                    + " in " + this + ": " + resource, predicate,
307                                                    this);
308                            }
309                            uris.add(uri);
310                    }
311                    return uris;
312            }
313    
314            public SortedSet<String> getPropertiesAsStrings(URI predicate)
315            throws UnexpectedPropertyException {
316                    SortedSet<String> strings = new TreeSet<String>();
317                    for (PropertyLiteral literal : getPropertiesAsLiterals(predicate)) {
318                            strings.add(literal.getLiteralValue());
319                    }
320                    return strings;
321            }
322    
323            public <PropertyType extends PropertyObject> SortedSet<PropertyType> getPropertiesOfType(
324                            URI predicate, Class<PropertyType> propertyType)
325                            throws UnexpectedPropertyException {
326                    SortedSet<PropertyType> properties = new TreeSet<PropertyType>(
327                                    new PropertyComparator());
328                    for (PropertyObject obj : getProperties().get(predicate)) {
329                            if (!propertyType.isInstance(obj)) {
330                                    throw new UnexpectedPropertyException("Not a " + propertyType
331                                                    + ": " + predicate + " in " + this, predicate, this);
332                            }
333                            properties.add(propertyType.cast(obj));
334                    }
335                    return properties;
336            }
337    
338            public PropertyObject getProperty(URI predicate)
339            throws PropertyNotFoundException, MultiplePropertiesException {
340                    PropertyObject foundProperty = null;
341                    // Could have checked set's size() - but it's
342                    for (PropertyObject obj : getProperties().get(predicate)) {
343                            if (foundProperty != null) {
344                                    throw new MultiplePropertiesException(predicate, this);
345                            }
346                            foundProperty = obj;
347                    }
348                    if (foundProperty == null) {
349                            throw new PropertyNotFoundException(predicate, this);
350                    }
351                    return foundProperty;
352            }

356                    throws UnexpectedPropertyException, PropertyNotFoundException, MultiplePropertiesException {
357                            return getPropertyOfType(predicate, PropertyList.class);
358            }

374            public PropertyResource getPropertyAsResource(URI predicate)
375                            throws UnexpectedPropertyException, PropertyNotFoundException,

378                                    PropertyResource.class);
379                    return propertyResource;

382            public URI getPropertyAsResourceURI(URI predicate)
383            throws UnexpectedPropertyException, PropertyNotFoundException,
384            MultiplePropertiesException {
385                    PropertyReference propertyResource = getPropertyOfType(predicate,
386                                    PropertyReference.class);
387                    URI uri = propertyResource.getResourceURI();
388                    if (uri == null) {
389                            throw new UnexpectedPropertyException(
390                                            "Resource property without URI for " + predicate + " in "
391                                            + this + ": " + propertyResource, predicate, this);
392                    }
393                    return uri;
394            }
395    
396            public String getPropertyAsString(URI predicate) throws PropertyException {
397                    PropertyLiteral propertyLiteral = getPropertyOfType(predicate,
398                                    PropertyLiteral.class);
399                    return propertyLiteral.getLiteralValue();
400            }
401    
402            public <PropertyType extends PropertyObject> PropertyType getPropertyOfType(
403                            URI predicate, Class<PropertyType> propertyType)
404            throws UnexpectedPropertyException, PropertyNotFoundException,
405            MultiplePropertiesException {
406                    PropertyObject propObj = getProperty(predicate);
407                    if (!propertyType.isInstance(propObj)) {
408                            throw new UnexpectedPropertyException("Not a " + propertyType
409                                            + ": " + predicate + " in " + this, predicate, this);
410                    }
411                    return propertyType.cast(propObj);
412            }

414            public final URI getTypeURI() {
415                    return typeURI;
416            }
417    
418            public final void setProperties(
419                            Map<URI, SortedSet<PropertyObject>> properties) {
420                    this.properties.clear();
421                    this.properties.putAll(properties);
422            }
423    
424            public final void setTypeURI(URI typeURI) {
425                    this.typeURI = typeURI;
426            }
427    
428            @Override
429            public String toString() {
430                    return "PropertyResource [getTypeURI()=" + getTypeURI()
431                    + ", getResourceURI()=" + getResourceURI() + "]";
432            }
433    
434            public boolean hasProperty(URI predicate) {
435                    if (! getProperties().containsKey(predicate)) {
436                            return false;
437                    }
438                    // Might be an empty set
439                    return ! getProperties().get(predicate).isEmpty();
440            }
