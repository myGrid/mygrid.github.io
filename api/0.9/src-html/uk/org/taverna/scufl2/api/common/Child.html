001    package uk.org.taverna.scufl2.api.common;
002    
003    /**
004     * A {@link WorkflowBean} that is the child of another <code>WorkflowBean</code>.
005     * 
006     * @author Alan R Williams
007     * @author Stian Soiland-Reyes
008     * 
009     * @param <T>
010     *            the type of <code>WorkflowBean</code> that this is a child of
011     */
012    public interface Child<T extends WorkflowBean> extends WorkflowBean {
013    
014            /**
015             * Returns the parent of this workflow bean, or <code>null</code> if it is orphan.
016             * 
017             * @return the parent of this workflow bean, or <code>null</code> if it is orphan
018             */
019            T getParent();
020    
021            /**
022             * Sets the parent of this workflow bean.
023             * <p>
024             * Setting the parent would normally also add the object to the relevant collection in the
025             * parent if it does not already exist there.
026             * <p>
027             * If the child has an existing, object-identity different parent, the child will first be
028             * removed from the parent collection if it exists there.
029             * <p>
030             * <strong>Note:</strong>If the child is {@link Named} the parent collection will be a
031             * {@link NamedSet}. This implicit insertion would overwrite any conflicting sibling with the
032             * same {@link Named#getName()} - to avoid this, add the child to the parent collection by using
033             * {@link NamedSet#addWithUniqueName(Named)} before setting the parent.
034             * <p>
035             * 
036             * @param parent the parent of this workflow bean
037             */
038            void setParent(T parent);
039    
040    }
