042    /**
043     * A <code>WorkflowBundleWriter</code> that writes a {@link WorkflowBundle} in Scufl2 Structure
044     * format.
045     */
046    public class StructureWriter implements WorkflowBundleWriter {
047    
048            private StringBuffer sb;
049            private URITools uriTools = new URITools();
050    
051    
052            private void append(Named named) {
053                    append(" '");
054                    append(escapeName(named.getName()));
055                    append("'");
056            }
057    
058            private void append(String string) {
059                    sb.append(string);
060            }
061    
062            private void append(URI uri) {
063                    sb.append(" <");
064                    sb.append(uri.toASCIIString());
065                    sb.append(">");
066            }
067    
068            private void appendPorts(int indent, Ported wf) {
069                    for (Port in : sorted(wf.getInputPorts())) {
070                            newLine(indent);
071                            append("In");
072                            append(in);
073                    }
074                    for (Port out : sorted(wf.getOutputPorts())) {
075                            newLine(indent);
076                            append("Out");
077                            append(out);
078                    }
079            }
080    
081            @SuppressWarnings("unchecked")
082            protected String bundleString(WorkflowBundle wb) {
083                    sb = new StringBuffer();
084                    append("WorkflowBundle");
085                    append(wb);
086    
087                    if (wb.getMainWorkflow() != null) {
088                            newLine(1);
089                            append("MainWorkflow");
090                            append(wb.getMainWorkflow());
091                    }
092                    for (Workflow wf : sorted(wb.getWorkflows())) {
093                            newLine(1);
094                            append("Workflow");
095                            append(wf);
096                            appendPorts(2, wf);
097                            for (Processor p : sorted(wf.getProcessors())) {
098                                    newLine(2);
099                                    append("Processor");
100                                    append(p);
101                                    appendPorts(3, p);
102                            }
103    
104                            // LINKS
105                            // We'll need to sort them afterwards
106                            List<String> links = new ArrayList<String>();
107                            for (DataLink dl : wf.getDataLinks()) {
108                                    links.add(datalink(dl.getReceivesFrom()) + " -> " + datalink(dl.getSendsTo()));
109                            }
110                            Collections.sort(links);
111                            if (!links.isEmpty()) {
112                                    newLine(2);
113                                    append("Links");
114                            }
115                            for (String link : links) {
116                                    newLine(3);
117                                    append(link);
118                            }
119    
120                            List<ControlLink> controlLinks = new ArrayList<ControlLink>(wf.getControlLinks());
121                            if (!controlLinks.isEmpty()) {
122                                    newLine(2);
123                                    append("Controls");
124                                    Collections.sort(controlLinks);
125                                    String link = "block {1} until {2} finish";
126                                    for (ControlLink controlLink : controlLinks) {
127                                            if (!(controlLink instanceof BlockingControlLink)) {
128                                                    // TODO
129                                                    continue;
130                                            }
131                                            BlockingControlLink blockingControlLink = (BlockingControlLink) controlLink;
132                                            newLine(3);
133                                            append("block");
134                                            append(blockingControlLink.getBlock());
135                                            append(" until");
136                                            append(blockingControlLink.getUntilFinished());
137                                            append(" finish");
138                                    }
139                            }
140    
141                    }
142    
143                    if (wb.getMainProfile() != null) {
144                            newLine(1);
145                            append("MainProfile");
146                            append(wb.getMainProfile());
147                    }
148    
149                    for (Profile p : sorted(wb.getProfiles())) {
150                            newLine(1);
151                            append("Profile");
152                            append(p);
153                            for (Activity a : sorted(p.getActivities())) {
154                                    newLine(2);
155                                    append("Activity");
156                                    append(a);
157                                    newLine(3);
158                                    append("Type");
159                                    append(a.getConfigurableType());
160                                    appendPorts(3, a);
161                            }
162                            for (ProcessorBinding pb : p.getProcessorBindings()) {
163                                    newLine(2);
164                                    append("ProcessorBinding");
165                                    append(pb);
166                                    newLine(3);
167                                    append("Activity");
168                                    append(pb.getBoundActivity());
169                                    newLine(3);
170                                    append("Processor");
171                                    String name = " '" + escapeName(pb.getBoundProcessor().getParent().getName());
172                                    name = name + ":" + escapeName(pb.getBoundProcessor().getName()) + "'";
173                                    append(name);
174    
175                                    List<String> links = new ArrayList<String>();
176                                    for (ProcessorInputPortBinding ip : pb.getInputPortBindings()) {
177                                            links.add("'" + escapeName(ip.getBoundProcessorPort().getName()) + "' -> '"
178                                                            + escapeName(ip.getBoundActivityPort().getName()) + "'");
179                                    }
180                                    Collections.sort(links);
181                                    if (!links.isEmpty()) {
182                                            newLine(3);
183                                            append("InputPortBindings");
184                                    }
185                                    for (String link : links) {
186                                            newLine(4);
187                                            append(link);
188                                    }
189    
190                                    links.clear();
191                                    for (ProcessorOutputPortBinding ip : pb.getOutputPortBindings()) {
192                                            // Note: opposite direction as for ProcessorInputPortBinding
193                                            links.add("'" + escapeName(ip.getBoundActivityPort().getName()) + "' -> '"
194                                                            + escapeName(ip.getBoundProcessorPort().getName()) + "'");
195                                    }
196                                    Collections.sort(links);
197                                    if (!links.isEmpty()) {
198                                            newLine(3);
199                                            append("OutputPortBindings");
200                                    }
201                                    for (String link : links) {
202                                            newLine(4);
203                                            append(link);
204                                    }
205                            }
206    
207                            for (Configuration config : p.getConfigurations()) {
208                                    newLine(2);
209                                    append("Configuration");
210                                    append(config);
211    
212                                    newLine(3);
213                                    if (config.getConfigurableType() != null) {
214                                            append("Type");
215                                            append(config.getConfigurableType());
216                                    }
217                                    newLine(3);
218                                    append("Configures");
219                                    if (config.getConfigures() instanceof Named) {                          
220                                            Named c = (Named) config.getConfigures();                                       
221                                            String cName = "'" + escapeName(c.getClass().getSimpleName().toLowerCase());
222                                            cName = cName + "/" + escapeName(c.getName()) + "'";
223                                            append(" " + cName);
224                                    } else {
225                                            URI configuredURI = uriTools.relativeUriForBean(config.getConfigures(), p);
226                                            append(" '" + configuredURI.toASCIIString() + "'");
227                                    }
228                                    
229                                    for (Entry<URI, SortedSet<PropertyObject>> prop : config.getPropertyResource()
230                                                    .getProperties().entrySet()) {
231                                            newLine(3);
232                                            append("Property");
233                                            append(prop.getKey());
234    
235                                            for (PropertyObject po : prop.getValue()) {
236                                                    if (po instanceof PropertyLiteral) {
237                                                            PropertyLiteral lit = (PropertyLiteral) po;
238                                                            newLine(4);
239                                                            append("'''");
240                                                            append(lit.getLiteralValue().replace("'''", "\\'\\'\\'"));
241                                                            append("'''");
242    
243                                                            // TODO: Handle literal types
244                                                    } else {
245                                                            // TODO: Handle other props, recursively
246                                                    }
247    
248                                            }
249                                    }
250                            }
251    
252                    }
253                    append("\n");
254                    //System.out.println(sb);
255                    return sb.toString();
256            }
257    
258            private String datalink(Port port) {
259                    StringBuffer s = new StringBuffer();
260                    s.append("'");
261                    if (port instanceof ProcessorPort) {
262                            ProcessorPort processorPort = (ProcessorPort) port;
263                            s.append(escapeName(processorPort.getParent().getName()));
264                            s.append(":");
265                    }
266                    s.append(escapeName(port.getName()));
267                    s.append("'");
268                    return s.toString();
269            }
270    
271            private String escapeName(String name) {
272                    return name.replace("\\", "\\\\").replace("'", "\\'").replace(":", "\\:")
273                    .replace("/", "\\/");
274            }
275    
276            @Override
277            public Set<String> getMediaTypes() {
278                    return Collections.singleton(TEXT_VND_TAVERNA_SCUFL2_STRUCTURE);
279            }
280    
281            private void newLine(int indentLevel) {
282                    sb.append("\n");
283                    for (int i = 0; i < indentLevel; i++) {
284                            sb.append("  ");
285                    }
286            }
287    
288            private <T extends Named> List<T> sorted(NamedSet<T> namedSet) {
289                    List<T> sorted = new ArrayList<T>();
290                    sorted.addAll(namedSet);
291                    Collections.sort(sorted, new Comparator<T>() {
292                            @Override
293                            public int compare(T o1, T o2) {
294                                    return o1.getName().compareTo(o2.getName());
295                            }
296                    });
297                    return sorted;
298            }
299    
300            @Override
301            public synchronized void writeBundle(WorkflowBundle wb, File destination, String mediaType)
302            throws IOException {
303                    destination.createNewFile();
304    
305                    BufferedOutputStream outputStream = new BufferedOutputStream(new FileOutputStream(
306                                    destination));
307                    writeBundle(wb, outputStream, mediaType);
308                    outputStream.close();
309    
310            }
311    
312            @Override
313            public void writeBundle(WorkflowBundle wfBundle, OutputStream output, String mediaType)
314            throws IOException {
315                    OutputStreamWriter writer = new OutputStreamWriter(output, "utf-8");
316                    writer.write(bundleString(wfBundle));
317                    writer.close();
318            }
