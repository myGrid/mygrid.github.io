001    package uk.org.taverna.scufl2.api.activity;
002    
003    import java.net.URI;
004    import java.util.ArrayList;
005    import java.util.List;
006    import java.util.Set;
007    
008    import uk.org.taverna.scufl2.api.common.AbstractNamedChild;
009    import uk.org.taverna.scufl2.api.common.Child;
010    import uk.org.taverna.scufl2.api.common.Configurable;
011    import uk.org.taverna.scufl2.api.common.NamedSet;
012    import uk.org.taverna.scufl2.api.common.Ported;
013    import uk.org.taverna.scufl2.api.common.Typed;
014    import uk.org.taverna.scufl2.api.common.Visitor;
015    import uk.org.taverna.scufl2.api.common.WorkflowBean;
016    import uk.org.taverna.scufl2.api.port.InputActivityPort;
017    import uk.org.taverna.scufl2.api.port.OutputActivityPort;
018    import uk.org.taverna.scufl2.api.profiles.Profile;
019    
020    /**
021     * An Activity specifies a way of implementing a {@link uk.org.taverna.scufl2.api.core.Processor Processor}
022     * within a {@link uk.org.taverna.scufl2.api.core.Workflow Workflow}.
023     * <p>
024     * When the Workflow is run, a particular Activity will be specified as bound to the Processor and
025     * {@link uk.org.taverna.scufl2.api.configurations.Configuration Configuration} information will be specified for the Activity.
026     * 
027     * @author Alan R Williams
028     * @author Stian Soiland-Reyes
029     */
030    public class Activity extends AbstractNamedChild implements Configurable, Child<Profile>, Typed,
031    Ported {
032    
033            private final NamedSet<InputActivityPort> inputPorts = new NamedSet<InputActivityPort>();
034            private final NamedSet<OutputActivityPort> outputPorts = new NamedSet<OutputActivityPort>();
035    
036            private URI type;
037            private Profile parent;
038    
039            /**
040             * Constructs an <code>Activity</code> with a random UUID as the name.
041             */
042            public Activity() {
043                    super();
044            }
045    
046            /**
047             * Constructs an <code>Activity</code> with the specified name.
048             * 
049             * @param name
050             *            the name of the Activity. <strong>Must not</strong> be <code>null</code>
051             *            or an empty String.
052             */
053            public Activity(String name) {
054                    super(name);
055            }
056    
057            @Override
058            public boolean accept(Visitor visitor) {
059                    if (visitor.visitEnter(this)) {
060                            List<Iterable<? extends WorkflowBean>> children = new ArrayList<Iterable<? extends WorkflowBean>>();
061                            children.add(getInputPorts());
062                            children.add(getOutputPorts());
063                            outer: for (Iterable<? extends WorkflowBean> it : children) {
064                                    for (WorkflowBean bean : it) {
065                                            if (!bean.accept(visitor)) {
066                                                    break outer;
067                                            }
068                                    }
069                            }
070                    }
071                    return visitor.visitLeave(this);
072            }
073    
074            /**
075             * Returns the type of the <code>Activity</code>.
076             * 
077             * @return the type of the <code>Activity</code>
078             */
079            @Override
080            public URI getConfigurableType() {
081                    return type;
082            }
083    
084            /**
085             * Returns the <code>InputActivityPort</code>s.
086             * 
087             * If there are no <code>InputActivityPort</code>s an empty set is returned.
088             * 
089             * @return the <code>InputActivityPort</code>s.
090             */
091            @Override
092            public NamedSet<InputActivityPort> getInputPorts() {
093                    return inputPorts;
094            }
095    
096            /**
097             * Returns the <code>OutputActivityPort</code>s.
098             * 
099             * If there are no <code>OutputActivityPort</code>s an empty set is returned.
100             * 
101             * @return the <code>OutputActivityPort</code>s.
102             */
103            @Override
104            public NamedSet<OutputActivityPort> getOutputPorts() {
105                    return outputPorts;
106            }
107    
108            @Override
109            public Profile getParent() {
110                    return parent;
111            }
112    
113            /**
114             * Sets the type of the <code>Activity</code>.
115             * 
116             * @param type the type of the <code>Activity</code>
117             */
118            @Override
119            public void setConfigurableType(URI type) {
120                    this.type = type;
121            }
122    
123            /**
124             * Set the <code>InputActivityPort</code>s to be the contents of the specified set.
125             * <p>
126             * <code>InputActivityPort</code>s can be added by using {@link #getInputPorts()}.add(inputPort).
127             * 
128             * @param inputPorts the <code>InputActivityPort</code>s. <strong>Must not</strong> be null
129             */
130            public void setInputPorts(Set<InputActivityPort> inputPorts) {
131                    this.inputPorts.clear();
132                    this.inputPorts.addAll(inputPorts);
133            }
134    
135            /**
136             * Set the <code>OutputActivityPort</code>s to be the contents of the specified set.
137             * <p>
138             * <code>OutputActivityPort</code>s can be added by using {@link #getOutputPorts()}.add(outputPort).
139             * 
140             * @param outputPorts the <code>OutputActivityPort</code>s. <strong>Must not</strong> be null
141             */
142            public void setOutputPorts(Set<OutputActivityPort> outputPorts) {
143                    this.outputPorts.clear();
144                    this.outputPorts.addAll(outputPorts);
145            }
146    
147            @Override
148            public void setParent(Profile parent) {
149                    if (this.parent != null && this.parent != parent) {
150                            this.parent.getActivities().remove(this);
151                    }
152                    this.parent = parent;
153                    if (parent != null) {
154                            parent.getActivities().add(this);
155                    }
156            }
157    
158            @Override
159            public String toString() {
160                    return "Activity " + getConfigurableType() + " \"" + getName() + '"';
161            }
162    }
