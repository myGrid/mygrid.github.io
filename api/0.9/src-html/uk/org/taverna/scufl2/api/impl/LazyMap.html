001    package uk.org.taverna.scufl2.api.impl;
002    
003    import java.util.Comparator;
004    import java.util.Map;
005    import java.util.SortedMap;
006    import java.util.TreeMap;
007    
008    /**
009     * A lazy TreeMap, inspired by org.apache.commons.collections.map.LazyMap
010     * <p>
011     * On {@link #get(Object)}, if a key is not found in the map,
012     * {@link #getDefault(Object)} will be called to create the value for the given
013     * key. This value is subsequently inserted into the map before being returned.
014     * <p>
015     * Call {@link #containsKey(Object)} to check if a key is in the map.
016     * <p>
017     * Implementations of this class must implement {@link #getDefault(Object)} to
018     * specify the default value to create.
019     *
020     * @author Stian Soiland-Reyes
021     *
022     * @param <KeyType>
023     *            Type of key
024     * @param <ValueType>
025     *            Type of value
026     */
027    public abstract class LazyMap<KeyType, ValueType> extends
028                    TreeMap<KeyType, ValueType> implements Map<KeyType, ValueType> {
029    
030            private static final long serialVersionUID = 3284689384208221667L;
031    
032            public LazyMap() {
033                    super();
034            }
035    
036            public LazyMap(Comparator<? super KeyType> comparator) {
037                    super(comparator);
038            }
039    
040            public LazyMap(Map<? extends KeyType, ? extends ValueType> m) {
041                    super(m);
042            }
043    
044            public LazyMap(SortedMap<KeyType, ? extends ValueType> m) {
045                    super(m);
046            }
047    
048    
049            @SuppressWarnings("unchecked")
050            @Override
051            public ValueType get(Object key) {
052                    ValueType value = super.get(key);
053                    if (value == null) {
054                            value = getDefault((KeyType) key);
055                            put((KeyType) key, value);
056                    }
057                    return value;
058            }
059    
060            public abstract ValueType getDefault(KeyType key);
061    }
