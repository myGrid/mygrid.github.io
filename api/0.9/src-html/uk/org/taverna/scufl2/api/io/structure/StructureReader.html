038    /**
039     * A <code>WorkflowBundleReader</code> that reads a {@link WorkflowBundle} in Scufl2 Structure
040     * format.
041     */
042    public class StructureReader implements WorkflowBundleReader {
043    
044            public enum Level {
045                    WorkflowBundle, Workflow, Processor, Activity, Links, Profile, Configuration, ProcessorBinding, OutputPortBindings, InputPortBindings, Property, Controls
046    
047            }
048    
049            private static final String ACTIVITY_SLASH = "activity/";
050    
051            public static final String TEXT_VND_TAVERNA_SCUFL2_STRUCTURE = "text/vnd.taverna.scufl2.structure";
052    
053            private WorkflowBundle wb;
054    
055            private Level level;
056    
057            private String mainWorkflow;
058    
059            private Workflow workflow;
060    
061            private Processor processor;
062    
063            private Activity activity;
064    
065            Pattern linkPattern = Pattern
066            .compile("'(.*[^\\\\])'\\s->\\s'(.*[^\\\\\\\\])'");
067    
068            Pattern blockPattern = Pattern
069            .compile("\\s*block\\s+'(.*[^\\\\])'\\s+until\\s+'(.*[^\\\\\\\\])'\\s+finish");
070    
071            private String mainProfile;
072    
073            private Profile profile;
074    
075            private Configuration configuration;
076    
077            private ProcessorBinding processorBinding;
078    
079            private URI propertyUri;
080    
081            private StringBuffer propertyString;
082    
083            private StringBuffer largeString;
084    
085            @Override
086            public Set<String> getMediaTypes() {
087                    return Collections.singleton(TEXT_VND_TAVERNA_SCUFL2_STRUCTURE);
088            }
089    
090            protected WorkflowBundle parse(InputStream is) throws ReaderException {
091    
092                    wb = new WorkflowBundle();
093                    Scanner scanner = new Scanner(is);
094                    try {
095                            while (scanner.hasNextLine()) {
096                                    parseLine(scanner.nextLine());
097                            }
098                    } finally {
099                            scanner.close();
100                    }
101    
102                    return wb;
103            }
104    
105            protected void parseLine(String nextLine) throws ReaderException {
106                    Scanner scanner = new Scanner(nextLine.trim());
107                    // allow any whitespace
108                    String next = scanner.next();
109    
110                    if (next.isEmpty()) {
111                            return;
112                    }
113                    if (next.equals("WorkflowBundle")) {
114                            level = Level.WorkflowBundle;
115                            String name = parseName(scanner);
116                            wb.setName(name);
117                            return;
118                    }
119                    if (next.equals("MainWorkflow")) {
120                            mainWorkflow = parseName(scanner);
121                            return;
122                    }
123                    if (next.equals("Workflow")) {
124                            level = Level.Workflow;
125                            workflow = new Workflow();
126                            String workflowName = parseName(scanner);
127                            workflow.setName(workflowName);
128                            wb.getWorkflows().add(workflow);
129                            if (workflowName.equals(mainWorkflow)) {
130                                    wb.setMainWorkflow(workflow);
131                            }
132                            return;
133                    }
134                    if (next.equals("In") || next.equals("Out")) {
135                            boolean in = next.equals("In");
136                            String portName = parseName(scanner);
137                            switch (level) {
138                                    case Workflow:
139                                            if (in) {
140                                                    new InputWorkflowPort(workflow, portName);
141                                            } else {
142                                                    new OutputWorkflowPort(workflow, portName);
143                                            }
144                                            break;
145                                    case Processor:
146                                            if (in) {
147                                                    new InputProcessorPort(processor, portName);
148                                            } else {
149                                                    new OutputProcessorPort(processor, portName);
150                                            }
151                                            break;
152                                    case Activity:
153                                            if (in) {
154                                                    new InputActivityPort(activity, portName);
155                                            } else {
156                                                    new OutputActivityPort(activity, portName);
157                                            }
158                                            break;
159                                    default:
160                                            throw new ReaderException("Unexpected " + next + " at level "
161                                                            + level);
162                            }
163                            return;
164                    }
165                    if (next.equals("Processor")
166                                    && (level == Level.Workflow || level == Level.Processor)) {
167                            level = Level.Processor;
168                            processor = new Processor();
169                            String processorName = parseName(scanner);
170                            processor.setName(processorName);
171                            processor.setParent(workflow);
172                            workflow.getProcessors().add(processor);
173                            return;
174                    }
175                    if (next.equals("Links")) {
176                            level = Level.Links;
177                            processor = null;
178                            return;
179                    }
180                    if (next.equals("Controls")) {
181                            level = Level.Controls;
182                            return;
183                    }
184                    if (next.equals("block")) {
185                            Matcher blockMatcher = blockPattern.matcher(nextLine);
186                            blockMatcher.find();
187                            String block = blockMatcher.group(1);
188                            String untilFinish = blockMatcher.group(2);
189    
190                            Processor blockProc = workflow.getProcessors().getByName(block);
191                            Processor untilFinishedProc = workflow.getProcessors().getByName(
192                                            untilFinish);
193                            new BlockingControlLink(blockProc, untilFinishedProc);
194                    }
195                    if (next.startsWith("'") && level.equals(Level.Links)) {
196                            Matcher linkMatcher = linkPattern.matcher(nextLine);
197                            linkMatcher.find();
198                            String firstLink = linkMatcher.group(1);
199                            String secondLink = linkMatcher.group(2);
200    
201                            SenderPort senderPort;
202                            if (firstLink.contains(":")) {
203                                    String[] procPort = firstLink.split(":");
204                                    Processor proc = workflow.getProcessors()
205                                    .getByName(procPort[0]);
206                                    senderPort = proc.getOutputPorts().getByName(procPort[1]);
207                            } else {
208                                    senderPort = workflow.getInputPorts().getByName(firstLink);
209                            }
210    
211                            ReceiverPort receiverPort;
212                            if (secondLink.contains(":")) {
213                                    String[] procPort = secondLink.split(":");
214                                    Processor proc = workflow.getProcessors()
215                                    .getByName(procPort[0]);
216                                    receiverPort = proc.getInputPorts().getByName(procPort[1]);
217                            } else {
218                                    receiverPort = workflow.getOutputPorts().getByName(secondLink);
219                            }
220    
221                            new DataLink(workflow, senderPort, receiverPort);
222                            return;
223                    }
224    
225                    if (next.equals("MainProfile")) {
226                            mainProfile = parseName(scanner);
227                            return;
228                    }
229                    if (next.equals("Profile")) {
230                            level = Level.Profile;
231                            profile = new Profile();
232                            String profileName = parseName(scanner);
233                            profile.setName(profileName);
234                            wb.getProfiles().add(profile);
235                            if (profileName.equals(mainProfile)) {
236                                    wb.setMainProfile(profile);
237                            }
238                            return;
239                    }
240                    if (next.equals("Activity")
241                                    && (level == Level.Profile || level == Level.Activity)) {
242                            level = Level.Activity;
243                            activity = new Activity();
244                            String activityName = parseName(scanner);
245                            activity.setName(activityName);
246                            profile.getActivities().add(activity);
247                            return;
248                    }
249                    if (next.equals("Type")) {
250                            URI uri = URI.create(nextLine.split("[<>]")[1]);
251                            if (level == Level.Activity) {
252                                    activity.setConfigurableType(uri);
253                            } else if (level == Level.Configuration) {
254                                    configuration.getPropertyResource().setTypeURI(uri);
255                            }
256                            return;
257                    }
258                    if (next.equals("ProcessorBinding")) {
259                            level = Level.ProcessorBinding;
260                            processorBinding = new ProcessorBinding();
261                            String bindingName = parseName(scanner);
262                            processorBinding.setName(bindingName);
263                            profile.getProcessorBindings().add(processorBinding);
264                            return;
265                    }
266                    if (next.equals("Activity") && level == Level.ProcessorBinding) {
267                            String activityName = parseName(scanner);
268                            Activity boundActivity = profile.getActivities().getByName(
269                                            activityName);
270                            processorBinding.setBoundActivity(boundActivity);
271                            return;
272                    }
273                    if (next.equals("Processor") && level == Level.ProcessorBinding) {
274                            String[] wfProcName = parseName(scanner).split(":");
275                            Workflow wf = wb.getWorkflows().getByName(wfProcName[0]);
276                            Processor boundProcessor = wf.getProcessors().getByName(
277                                            wfProcName[1]);
278                            processorBinding.setBoundProcessor(boundProcessor);
279                            return;
280                    }
281                    if (next.equals("InputPortBindings")) {
282                            level = Level.InputPortBindings;
283                            return;
284                    }
285                    if (next.equals("OutputPortBindings")) {
286                            level = Level.OutputPortBindings;
287                            return;
288                    }
289    
290                    if (next.startsWith("'")
291                                    && (level == Level.InputPortBindings || level == Level.OutputPortBindings)) {
292                            Matcher linkMatcher = linkPattern.matcher(nextLine);
293                            linkMatcher.find();
294                            String firstLink = linkMatcher.group(1);
295                            String secondLink = linkMatcher.group(2);
296                            if (level == Level.InputPortBindings) {
297                                    InputProcessorPort processorPort = processorBinding
298                                    .getBoundProcessor().getInputPorts()
299                                    .getByName(firstLink);
300                                    InputActivityPort activityPort = processorBinding
301                                    .getBoundActivity().getInputPorts()
302                                    .getByName(secondLink);
303                                    new ProcessorInputPortBinding(processorBinding, processorPort,
304                                                    activityPort);
305                            } else {
306                                    OutputActivityPort activityPort = processorBinding
307                                    .getBoundActivity().getOutputPorts()
308                                    .getByName(firstLink);
309                                    OutputProcessorPort processorPort = processorBinding
310                                    .getBoundProcessor().getOutputPorts()
311                                    .getByName(secondLink);
312                                    new ProcessorOutputPortBinding(processorBinding, activityPort,
313                                                    processorPort);
314                            }
315                            return;
316                    }
317                    if (next.equals("Configuration")) {
318                            level = Level.Configuration;
319                            configuration = new Configuration();
320                            profile.getConfigurations().add(configuration);
321                            String configName = parseName(scanner);
322                            configuration.setName(configName);
323                            return;
324                    }
325                    if (next.equals("Configures")) {
326                            String configures = parseName(scanner);
327                            if (configures.startsWith(ACTIVITY_SLASH)) {
328                                    String activityName = configures.substring(
329                                                    ACTIVITY_SLASH.length(), configures.length());
330                                    activity = profile.getActivities().getByName(activityName);
331                                    configuration.setConfigures(activity);
332                                    return;
333                            } else {
334                                    throw new UnsupportedOperationException("Unknown Configures "
335                                                    + configures);
336                            }
337                    }
338                    if (next.equals("Property")) {
339                            propertyUri = URI.create(nextLine.split("[<>]")[1]);
340                            level = Level.Property;
341                            return;
342                    }
343                    if (level == Level.Property) {
344                            boolean finished = false;
345                            String[] split = nextLine.split("'''", 3);
346                            if (split.length == 1) {
347                                    largeString.append(split[0]);
348                                    largeString.append('\n');
349                            } else if (next.startsWith("'''")) {
350                                    largeString = new StringBuffer();
351                                    largeString.append(split[1]);
352                                    if (split.length == 3) {
353                                            // It was a one-liner
354                                            finished = true;
355                                    } else {
356                                            largeString.append('\n');
357                                    }
358                            } else {
359                                    largeString.append(split[0]);
360                                    // Assuming length is 2
361                                    finished = true;
362                            }
363    
364                            if (finished) {
365                                    configuration.getPropertyResource().addPropertyAsString(
366                                                    propertyUri, largeString.toString());
367                                    largeString = null;
368                                    propertyUri = null;
369                                    level = Level.Configuration;
370                            }
371    
372                    }
373    
374            }
375    
376    
377            private String parseName(Scanner scanner) {
378                    String name = scanner.findInLine("'(.*[^\\\\])'");
379                    return name.substring(1, name.length() - 1);
380            }
381    
382            @Override
383            public WorkflowBundle readBundle(File bundleFile, String mediaType)
384            throws IOException, ReaderException {
385                    BufferedInputStream is = new BufferedInputStream(new FileInputStream(
386                                    bundleFile));
387                    try {
388                            return parse(is);
389                    } finally {
390                            is.close();
391                    }
392            }
393    
394            @Override
395            public WorkflowBundle readBundle(InputStream inputStream, String mediaType)
396            throws IOException, ReaderException {
397                    return parse(inputStream);
398    
399            }
400    
401            @Override
402            public String guessMediaTypeForSignature(byte[] firstBytes) {
403                    if (new String(firstBytes, Charset.forName("ISO-8859-1")).contains("WorkflowBundle '")) {
404                            return TEXT_VND_TAVERNA_SCUFL2_STRUCTURE;
405                    }
406                    return null;
407            }
