004    package uk.org.taverna.scufl2.api.configurations;
005    
006    import java.net.URI;
007    
008    import uk.org.taverna.scufl2.api.activity.Activity;
009    import uk.org.taverna.scufl2.api.common.AbstractNamedChild;
010    import uk.org.taverna.scufl2.api.common.Child;
011    import uk.org.taverna.scufl2.api.common.Configurable;
012    import uk.org.taverna.scufl2.api.common.Typed;
013    import uk.org.taverna.scufl2.api.common.Visitor;
014    import uk.org.taverna.scufl2.api.common.WorkflowBean;
015    import uk.org.taverna.scufl2.api.core.Processor;
016    import uk.org.taverna.scufl2.api.core.Workflow;
017    import uk.org.taverna.scufl2.api.dispatchstack.DispatchStackLayer;
018    import uk.org.taverna.scufl2.api.port.Port;
019    import uk.org.taverna.scufl2.api.profiles.Profile;
020    import uk.org.taverna.scufl2.api.property.PropertyResource;

022    /**
023     * Configuration of a {@link Configurable} workflow bean.
024     * <p>
025     * A configuration is activated by a {@link Profile}, and provides a link to the
026     * {@link #getPropertyResource()} containing the properties to configure the bean, like an
027     * {@link Activity}.
028     * <p>
029     * A configuration is of a certain (RDF) <strong>type</strong>, as defined by
030     * {@link PropertyResource#getTypeURI()} on the - which determines which properties are required and
031     * optional. For instance, the type
032     * <code>http://ns.taverna.org.uk/2010/activity/wsdl/ConfigType</code> requires the property
033     * <code>http://ns.taverna.org.uk/2010/activity/wsdl/operation</code>.
034     * <p>
035     * These requirements are described in the {@link ConfigurationDefinition} which
036     * {@link ConfigurationDefinition#getConfigurableType()} matches this configuration's
037     * {@link #getTypeURI()}. <strong>TODO: Where are the ConfigurationDefinitions found?</strong>
038     * 
039     * @author Alan R Williams
040     * @author Stian Soiland-Reyes
041     * 
042     */
043    public class Configuration extends AbstractNamedChild implements Child<Profile>, Typed {
044            private Configurable configures;
045            private Profile parent;
046            private PropertyResource propertyResource = new PropertyResource();
047    
048            /**
049             * Constructs a <code>Configuration</code> with a random UUID as the name.
050             */
051            public Configuration() {
052                    super();
053            }
054    
055            /**
056             * Constructs a <code>Configuration</code> with the specified name.
057             * 
058             * @param name
059             *            the name of the <code>Configuration</code>. <strong>Must not</strong> be
060             *            <code>null</code> or an empty String.
061             */
062            public Configuration(String name) {
063                    super(name);
064            }
065    
066            @Override
067            public boolean accept(Visitor visitor) {
068                    if (visitor.visitEnter(this) && getPropertyResource() != null) {
069                            getPropertyResource().accept(visitor);
070                    }
071                    return visitor.visitLeave(this);
072            }
073    
074            /**
075             * Returns the type of the <code>Configuration</code>.
076             * 
077             * @return the type of the <code>Configuration</code>
078             */
079            @Override
080            public URI getConfigurableType() {
081                    return getPropertyResource().getTypeURI();
082            }
083    
084            /**
085             * Returns the {@link Configurable} workflow bean that is configured. Typically an
086             * {@link Activity} or {@link DispatchStackLayer}, but in theory also {@link Processor},
087             * {@link Workflow} and {@link Port} can be configured.
088             * 
089             * @return the <code>Configurable</code> <code>WorkflowBean</code> that is configured
090             */
091            public Configurable getConfigures() {
092                    return configures;
093            }
094    
095            @Override
096            public Profile getParent() {
097                    return parent;
098            }
099    
100            /**
101             * Returns the underlying {@link PropertyResource} which contains the properties set by this
102             * configuration.
103             * 
104             * @return the backing {@link PropertyResource}.
105             */
106            public PropertyResource getPropertyResource() {
107                    return propertyResource;
108            }
109    
110            /**
111             * Sets the type of the <code>Configuration</code>.
112             * 
113             * @param type
114             *            the type of the <code>Configuration</code>.
115             */
116            @Override
117            public void setConfigurableType(URI type) {
118                    getPropertyResource().setTypeURI(type);
119            }
120    
121            /**
122             * Sets the {@link Configurable} {@link WorkflowBean} that is configured.
123             * 
124             * @param configurable
125             *            the <code>Configurable</code> <code>WorkflowBean</code> that is configured
126             */
127            public void setConfigures(Configurable configurable) {
128                    configures = configurable;
129            }
130    
131            @Override
132            public void setParent(Profile parent) {
133                    if (this.parent != null && this.parent != parent) {
134                            this.parent.getConfigurations().remove(this);
135                    }
136                    this.parent = parent;
137                    if (parent != null) {
138                            parent.getConfigurations().add(this);
139                    }
140    
141            }
142    
143            /**
144             * Sets the underlying {@link PropertyResource} which contains the properties set by this
145             * configuration.
146             * 
147             * @param propertyResource
148             *            the underlying <code>PropertyResource</code> which contains the properties set by
149             *            this configuration
150             */
151            public void setPropertyResource(PropertyResource propertyResource) {
152                    this.propertyResource = propertyResource;
153            }
154    
155    }
