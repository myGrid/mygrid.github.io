001    package uk.org.taverna.scufl2.api.configurations;
002    
003    import java.net.URI;
004    import java.util.LinkedHashSet;
005    
006    import uk.org.taverna.scufl2.api.property.PropertyReference;
007    
008    /**
009     * The definition of a {@link PropertyReference}.
010     *
011     * @author Stian Soiland-Reyes
012     *
013     */
014    public class PropertyReferenceDefinition extends PropertyDefinition {
015    
016            private LinkedHashSet<URI> options;
017    
018            /**
019             * Create a definition of a {@link PropertyReference}.
020             */
021            public PropertyReferenceDefinition() {
022                    options = new LinkedHashSet<URI>();
023            }
024    
025            /**
026             * Create a definition of an {@link PropertyReference}.
027             * 
028             * @param predicate
029             *            the URI identifying the <code>PropertyReference</code> that
030             *            this class defines
031             * @param name
032             *            the name of the <code>PropertyReference</code>
033             * @param label
034             *            a human readable label for the <code>PropertyReference</code>
035             * @param description
036             *            a description of the <code>PropertyReference</code>
037             * @param required
038             *            whether the <code>PropertyReference</code> is mandatory
039             * @param multiple
040             *            whether there can be multiple instances of the
041             *            <code>PropertyReference</code>
042             * @param ordered
043             *            whether the order of multiple instances of the
044             *            <code>Property</code> is significant
045             */
046            public PropertyReferenceDefinition(URI predicate, String name,
047                            String label, String description, boolean required,
048                            boolean multiple, boolean ordered) {
049                    super(predicate, name, label, description, required, multiple, ordered);
050                    options = new LinkedHashSet<URI>();
051            }
052    
053            /**
054             * Create a definition of an {@link PropertyReference}.
055             * 
056             * @param predicate
057             *            the URI identifying the <code>PropertyReference</code> that
058             *            this class defines
059             * @param name
060             *            the name of the <code>PropertyReference</code>
061             * @param label
062             *            a human readable label for the <code>PropertyReference</code>
063             * @param description
064             *            a description of the <code>PropertyReference</code>
065             * @param required
066             *            whether the <code>PropertyReference</code> is mandatory
067             * @param multiple
068             *            whether there can be multiple instances of the
069             *            <code>PropertyReference</code>
070             * @param ordered
071             *            whether the order of multiple instances of the
072             *            <code>Property</code> is significant
073             * @param options
074             *            Set of options
075             */
076            public PropertyReferenceDefinition(URI predicate, String name,
077                            String label, String description, boolean required,
078                            boolean multiple, boolean ordered, LinkedHashSet<URI> options) {
079                    super(predicate, name, label, description, required, multiple, ordered);
080                    this.options = options;
081            }
082    
083            public LinkedHashSet<URI> getOptions() {
084                    return options;
085            }
086    
087            public void setOptions(LinkedHashSet<URI> options) {
088                    this.options = options;
089            }
090    
091            @Override
092            protected String toString(String indent) {
093                    StringBuilder sb = new StringBuilder();
094                    sb.append(indent);
095                    sb.append("PropertyReferenceDefinition ");
096                    if (getPredicate() != null) sb.append(getPredicate());
097                    sb.append("\n");
098                    sb.append(indent + "  label=" + getLabel() + ", description=" + getDescription() + "\n");
099                    sb.append(indent + "  required=" + isRequired() + ", multiple=" + isMultiple()+ ", ordered=" + isOrdered() + "\n");
100                    if (getOptions().size() > 0) {
101                            sb.append("  options=" + getOptions());                 
102                    }
103                    return sb.toString();
104            }
105    
106    }
