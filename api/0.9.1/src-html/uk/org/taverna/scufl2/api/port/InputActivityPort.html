004    package uk.org.taverna.scufl2.api.port;
005    
006    import uk.org.taverna.scufl2.api.activity.Activity;
007    
008    /**
009     * An <code>InputActivityPort</code> is a <Port> that inputs data to an {@link Activity}.
010     * 
011     * @author Alan R Williams
012     */
013    public class InputActivityPort extends AbstractDepthPort implements ActivityPort, InputPort {
014    
015            private Activity parent;
016    
017            /**
018             * Constructs an <code>InputActivityPort</code> with a random UUID as the name.
019             */
020            public InputActivityPort() {
021            }
022    
023            /**
024             * Constructs an <code>InputPort</code> for the specified <code>Activity</code> with the
025             * specified name.
026             * <p>
027             * The <code>InputPort</code> is added to the <code>Activity</code> (if the
028             * <code>Activity</code> is not <code>null</code>).
029             * 
030             * @param activity
031             *            the <code>Activity</code> to add this <code>Port</code> to. Can be
032             *            <code>null</code>
033             * @param name
034             *            the name of the <code>Port</code>. <strong>Must not</strong> be <code>null</code>
035             *            or an empty String.
036             */
037            public InputActivityPort(Activity activity, String portName) {
038                    setParent(activity);
039                    setName(portName);
040            }
041    
042            @Override
043            public Activity getParent() {
044                    return parent;
045            }
046    
047            @Override
048            public void setParent(Activity parent) {
049                    if (this.parent != null && this.parent != parent) {
050                            this.parent.getInputPorts().remove(this);
051                    }
052                    this.parent = parent;
053                    if (parent != null) {
054                            parent.getInputPorts().add(this);
055                    }
056            }
057    
058    }
