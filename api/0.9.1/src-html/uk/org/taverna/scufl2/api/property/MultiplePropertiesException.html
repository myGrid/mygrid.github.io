021    package uk.org.taverna.scufl2.api.property;
022    
023    import java.net.URI;
024    
025    /**
026     * Thrown when more than one {@link PropertyObject} was found, buy only one property was expected.
027     * 
028     * @see PropertyResource#getProperty(java.net.URI)
029     * @see PropertyResource#getPropertyAsString(java.net.URI)
030     * @see PropertyResource#getPropertyAsResourceURI(java.net.URI)
031     * 
032     * @author Stian Soiland-Reyes
033     */
034    public class MultiplePropertiesException extends PropertyException {
035    
036            private static final long serialVersionUID = 2801964990757296381L;
037    
038            public MultiplePropertiesException(URI predicate,
039                            PropertyResource propertyResource) {
040                    super("More than one property " + predicate + " found in "
041                                    + propertyResource, predicate, propertyResource);
042            }
043    
044    }
