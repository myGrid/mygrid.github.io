050            /**
051             * Compare {@link ProcessorBinding}s by their {@link ProcessorBinding#getActivityPosition()}.
052             * 
053             * Note: this comparator imposes orderings that are inconsistent with equals.
054             * 
055             * @author Stian Soiland-Reyes
056             */
057            public static class BindingComparator implements Comparator<ProcessorBinding> {
058    
059                    @Override
060                    public int compare(ProcessorBinding o1, ProcessorBinding o2) {
061                            return o1.getActivityPosition() - o2.getActivityPosition();
062                    }
063    
064            }

066            /**
067             * Returns the {@link Configuration} for a {@link Configurable} in the given {@link Profile}.
068             * 
069             * @param configurable
070             *            the <code>Configurable</code> to find a <code>Configuration</code> for
071             * @param profile
072             *            the <code>Profile</code> to look for the <code>Configuration</code> in
073             * @return the <code>Configuration</code> for a <code>Configurable</code> in the given
074             *         <code>Profile</code>
075             */
076            public Configuration configurationFor(Configurable configurable, Profile profile) {
077                    List<Configuration> configurations = configurationsFor(configurable, profile);
078                    if (configurations.isEmpty()) {
079                            throw new IndexOutOfBoundsException("Could not find configuration for " + configurable);
080                    }
081                    if (configurations.size() > 1) {
082                            throw new IllegalStateException("More than one configuration for " + configurable);
083                    }
084                    return configurations.get(0);
085            }

087            public Configuration configurationForActivityBoundToProcessor(Processor concat, Profile profile) {
088                    ProcessorBinding binding = processorBindingForProcessor(concat, profile);
089                    Configuration config = configurationFor(binding.getBoundActivity(), profile);
090                    return config;
091            }

093            /**
094             * Returns the list of {@link Configuration Configurations} for a {@link Configurable} in the
095             * given {@link Profile}.
096             * 
097             * @param configurable
098             *            the <code>Configurable</code> to find a <code>Configuration</code> for
099             * @param profile
100             *            the <code>Profile</code> to look for the <code>Configuration</code> in
101             * @return the list of <code>Configurations</code> for a <code>Configurable</code> in the given
102             *         <code>Profile</code>
103             */
104            @SuppressWarnings("unchecked")
105            public List<Configuration> configurationsFor(Configurable configurable, Profile profile) {
106                    List<Configuration> configurations = new ArrayList<Configuration>();
107                    for (Configuration config : profile.getConfigurations()) {
108                            if (config.getConfigures().equals(configurable)) {
109                                    configurations.add(config);
110                            }
111                    }
112                    // Collections.sort(configurations);
113                    return configurations;
114            }

116            public List<BlockingControlLink> controlLinksBlocking(Processor blocked) {
117                    List<BlockingControlLink> controlLinks = new ArrayList<BlockingControlLink>();
118                    for (ControlLink link : blocked.getParent().getControlLinks()) {
119                            if (!(link instanceof BlockingControlLink)) {
120                                    continue;
121                            }
122                            BlockingControlLink blockingControlLink = (BlockingControlLink) link;
123                            if (blockingControlLink.getBlock().equals(blocked)) {
124                                    controlLinks.add(blockingControlLink);
125                            }
126                    }
127                    Collections.sort(controlLinks);
128                    return controlLinks;
129            }

175            public <T extends WorkflowBean> T findParent(Class<T> parentClass, Child<?> child) {
176                    WorkflowBean parent = child.getParent();
177                    if (parent == null) {
178                            return null;
179                    }
180                    if (parentClass.isAssignableFrom(parent.getClass())) {
181                            @SuppressWarnings("unchecked")
182                            T foundParent = (T) parent;
183                            return foundParent;
184                    }
185                    if (parent instanceof Child) {
186                            return findParent(parentClass, (Child<?>) parent);
187                    }
188                    return null;
189    
190            }

192            public PropertyResource portDefinitionFor(ActivityPort activityPort, Profile profile) throws PropertyException {
193                    Configuration actConfig = configurationFor(activityPort.getParent(), profile);
194                    
195                    URI portURI = uriTools.uriForBean(activityPort);
196                    URI configURI = uriTools.uriForBean(actConfig);
197    
198                    
199                    URI portDefinition;
200                    URI definesPort;
201                    if (activityPort instanceof InputPort) {
202                            portDefinition = PORT_DEFINITION.resolve("#inputPortDefinition");
203                            definesPort = PORT_DEFINITION.resolve("#definesInputPort");
204                    } else {
205                            portDefinition = PORT_DEFINITION.resolve("#outputPortDefinition");
206                            definesPort = PORT_DEFINITION.resolve("#definesOutputPort");
207                    }
208                    for (PropertyResource portDef : 
209                            actConfig.getPropertyResource().getPropertiesAsResources(portDefinition)) {                     
210                            URI portDefURI = portDef.getPropertyAsResourceURI(definesPort);
211                            // We'll compare the URIs as absolute URIs - but portDefURI is most likely relative
212                            // to the config
213                            if (configURI.resolve(portDefURI).equals(portURI)) {
214                                    return portDef;
215                            }
216                    }
217                    return null;
218                    
219            }

228                    }

232            public List<ProcessorBinding> processorBindingsForProcessor(Processor processor, Profile profile) {
233                    List<ProcessorBinding> bindings = new ArrayList<ProcessorBinding>();
234                    for (ProcessorBinding pb : profile.getProcessorBindings()) {
235                            if (pb.getBoundProcessor().equals(processor)) {
236                                    bindings.add(pb);
237                            }
238                    }
239                    Collections.sort(bindings, new BindingComparator());
240                    return bindings;
241            }

265            @SuppressWarnings("rawtypes")
266            protected ProcessorPortBinding processorPortBindingForPortInternal(Port port, Profile profile) {
267    
268                    List<ProcessorBinding> processorBindings;
269                    if (port instanceof ProcessorPort) {
270                            ProcessorPort processorPort = (ProcessorPort) port;
271                            processorBindings = processorBindingsForProcessor(processorPort.getParent(), profile);
272                    } else if (port instanceof ActivityPort) {
273                            ActivityPort activityPort = (ActivityPort) port;
274                            processorBindings = processorBindingsToActivity(activityPort.getParent());
275                    } else {
276                            throw new IllegalArgumentException("Port must be a ProcessorPort or ActivityPort");
277                    }
278                    for (ProcessorBinding procBinding : processorBindings) {
279                            Set<? extends ProcessorPortBinding> portBindings;
280                            if (port instanceof InputPort) {
281                                    portBindings = procBinding.getInputPortBindings();
282                            } else {
283                                    portBindings = procBinding.getOutputPortBindings();
284                            }
285                            for (ProcessorPortBinding portBinding : portBindings) {
286                                    if (port instanceof ProcessorPort
287                                                    && portBinding.getBoundProcessorPort().equals(port)) {
288                                            return portBinding;
289                                    }
290                                    if (port instanceof ActivityPort && portBinding.getBoundActivityPort().equals(port)) {
291                                            return portBinding;
292                                    }
293                            }
294                    }
295                    return null;
296            }

298            public void setParents(WorkflowBundle bundle) {
299                    bundle.accept(new VisitorWithPath() {
300                            @Override
301                            public boolean visit() {
302                                    WorkflowBean node = getCurrentNode();
303                                    if (node instanceof Child) {
304                                            Child child = (Child) node;
305                                            WorkflowBean parent = getCurrentPath().peek();
306                                            if (child.getParent() != parent) {
307                                                    child.setParent(parent);
308                                            }
309                                    }
310                                    return true;
311                            }
312    
313                    });
314            }

316            public DispatchStackLayer dispatchStackByType(Processor processor, URI type) {
317                    DispatchStackLayer candidate = null;
318                    for (DispatchStackLayer layer : processor.getDispatchStack()) {
319                            if (layer.getConfigurableType().equals(type)) {
320                                    if (candidate != null) {
321                                            throw new IllegalStateException("Found multiple dispatch stack layers of type "
322                                                            + type + " in " + processor);
323                                    }
324                                    candidate = layer;
325                            }
326                    }
327                    return candidate;
328                    
329            }
