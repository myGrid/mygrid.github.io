021    package uk.org.taverna.scufl2.api.configurations;
022    
023    import java.net.URI;
024    
025    import uk.org.taverna.scufl2.api.property.PropertyObject;
026    
027    /**
028     * Abstract definition of a {@link PropertyObject}.
029     *
030     * @author David Withers
031     */
032    public abstract class PropertyDefinition {
033    
034            private URI predicate;
035            private String name, label, description;
036            private boolean required, multiple, ordered;
037    
038    
039            /**
040             * Creates a PropertyDefinition that defines the attributes of a {@link PropertyObject}.
041             */
042            public PropertyDefinition() {
043            }
044    
045            /**
046             * Creates a PropertyDefinition that defines the attributes of a
047             * {@link PropertyObject}.
048             * 
049             * @param predicate
050             *            the URI identifying the <code>Property</code> that this class
051             *            defines
052             * @param name
053             *            the name of the <code>Property</code>
054             * @param label
055             *            a human readable label for the <code>Property</code>
056             * @param description
057             *            a description of the <code>Property</code>
058             * @param required
059             *            whether the <code>Property</code> is mandatory
060             * @param multiple
061             *            whether there can be multiple instances of the
062             *            <code>Property</code>
063             * @param ordered
064             *            whether the order of multiple instances of the
065             *            <code>Property</code> is significant
066             * @param options
067             */
068            public PropertyDefinition(URI predicate, String name, String label, String description,
069     boolean required, boolean multiple,
070                            boolean ordered) {
071                    this.predicate = predicate;
072                    this.name = name;
073                    this.label = label;
074                    this.description = description;
075                    this.required = required;
076                    this.multiple = multiple;
077                    this.ordered = ordered;
078            }
079    
080            @Override
081            public boolean equals(Object obj) {
082                    if (this == obj) {
083                            return true;
084                    }
085                    if (obj == null) {
086                            return false;
087                    }
088                    if (getClass() != obj.getClass()) {
089                            return false;
090                    }
091                    PropertyDefinition other = (PropertyDefinition) obj;
092                    if (description == null) {
093                            if (other.description != null) {
094                                    return false;
095                            }
096                    } else if (!description.equals(other.description)) {
097                            return false;
098                    }
099                    if (label == null) {
100                            if (other.label != null) {
101                                    return false;
102                            }
103                    } else if (!label.equals(other.label)) {
104                            return false;
105                    }
106                    if (multiple != other.multiple) {
107                            return false;
108                    }
109                    if (name == null) {
110                            if (other.name != null) {
111                                    return false;
112                            }
113                    } else if (!name.equals(other.name)) {
114                            return false;
115                    }
116                    if (ordered != other.ordered) {
117                            return false;
118                    }
119                    if (predicate == null) {
120                            if (other.predicate != null) {
121                                    return false;
122                            }
123                    } else if (!predicate.equals(other.predicate)) {
124                            return false;
125                    }
126                    if (required != other.required) {
127                            return false;
128                    }
129                    return true;
130            }
131    
132            /**
133             * Returns a description of the <code>Property</code>.
134             *
135             * @return a description of the <code>Property</code>
136             */
137            public String getDescription() {
138                    return description;
139            }
140    
141            /**
142             * Returns a human readable label for the <code>Property</code>.
143             *
144             * @return a human readable label for the <code>Property</code>
145             */
146            public String getLabel() {
147                    return label;
148            }
149    
150            /**
151             * Returns the name of the <code>Property</code>.
152             *
153             * @return the name of the <code>Property</code>
154             */
155            public String getName() {
156                    return name;
157            }
158    
159            /**
160             * Returns the URI identifying the <code>Property</code> that this class defines.
161             *
162             * @return the URI identifying the <code>Property</code> that this class defines
163             */
164            public URI getPredicate() {
165                    return predicate;
166            }
167    
168            @Override
169            public int hashCode() {
170                    final int prime = 31;
171                    int result = 1;
172                    result = prime * result + (description == null ? 0 : description.hashCode());
173                    result = prime * result + (label == null ? 0 : label.hashCode());
174                    result = prime * result + (multiple ? 1231 : 1237);
175                    result = prime * result + (name == null ? 0 : name.hashCode());
176                    result = prime * result + (ordered ? 1231 : 1237);
177                    result = prime * result + (predicate == null ? 0 : predicate.hashCode());
178                    result = prime * result + (required ? 1231 : 1237);
179                    return result;
180            }
181    
182            /**
183             * Returns true if there can be multiple instances of the <code>Property</code>.
184             *
185             * @return true if there can be multiple instances of the <code>Property</code>
186             */
187            public boolean isMultiple() {
188                    return multiple;
189            }
190    
191            /**
192             * Returns true if the order of multiple instances of the <code>Property</code> is significant.
193             *
194             * @return true if the order of multiple instances of the <code>Property</code> is significant
195             */
196            public boolean isOrdered() {
197                    return ordered;
198            }
199    
200            /**
201             * Returns true if the <code>Property</code> is mandatory.
202             *
203             * @return true if the <code>Property</code> is mandatory
204             */
205            public boolean isRequired() {
206                    return required;
207            }
208    
209            /**
210             * Sets a description of the <code>Property</code>.
211             *
212             * @param description a description of the <code>Property</code>
213             */
214            public void setDescription(String description) {
215                    this.description = description;
216            }
217    
218            /**
219             * Sets a human readable label for the <code>Property</code>.
220             *
221             * @param label a human readable label for the <code>Property</code>
222             */
223            public void setLabel(String label) {
224                    this.label = label;
225            }
226    
227            /**
228             * Sets whether there can be multiple instances of the <code>Property</code>.
229             *
230             * @param multiple whether there can be multiple instances of the <code>Property</code>
231             */
232            public void setMultiple(boolean multiple) {
233                    this.multiple = multiple;
234            }
235    
236            /**
237             * Sets the name of the <code>Property</code>.
238             *
239             * @param name the name of the <code>Property</code>
240             */
241            public void setName(String name) {
242                    this.name = name;
243            }
244    
245            /**
246             * Sets whether the order of multiple instances of the <code>Property</code> is significant.
247             *
248             * @param ordered the order of multiple instances of the <code>Property</code> is significant
249             */
250            public void setOrdered(boolean ordered) {
251                    this.ordered = ordered;
252            }
253    
254    
255    
256            /**
257             * Sets the URI identifying the <code>Property</code> that this class defines.
258             *
259             * @param predicate the URI identifying the <code>Property</code> that this class defines
260             */
261            public void setPredicate(URI predicate) {
262                    this.predicate = predicate;
263            }
264    
265            /**
266             * Sets whether the <code>Property</code> is mandatory.
267             *
268             * @param required whether the <code>Property</code> is mandatory
269             */
270            public void setRequired(boolean required) {
271                    this.required = required;
272            }
273    
274            @Override
275            public String toString() {
276                    return toString("");
277            }
278    
279            protected abstract String toString(String indent);
280    
281    }
