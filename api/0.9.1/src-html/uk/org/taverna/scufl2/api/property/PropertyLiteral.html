028    /**
029     * A {@link PropertyObject} representing a literal, like a string.
030     * <p>
031     * A literal always has a {@link #getLiteralValue()} and a {@link #getLiteralType()}. Note that RDF
032     * literal string's "language" is not yet supported.
033     * <p>
034     * Plain old strings should have the default {@link #setLiteralType(URI)} of {@link #XSD_STRING}.
035     * For known XSD types like {@value #XSD_NONNEGATIVEINTEGER} and {@link #XSD_NORMALIZEDSTRING}, the
036     * corresponding {@link #setLiteralType(URI)} should be set.
037     * <p>
038     * Some primitive Java types can be set directly by using constructors like
039     * {@link #PropertyLiteral(double)} and {@link #PropertyLiteral(boolean)}, and retrieved using
040     * methods like {@link #getLiteralValueAsDouble()} and {@link #getLiteralValueAsBoolean()}.
041     * <p>
042     * See {@link DatatypeConverter} to convert other XML types not covered by this class.
043     * <p>
044     * Values which are XML should have the type {@value #XML_LITERAL} set to avoid double-escaped serialisation.
045     * 
046     * @see DatatypeConverter
047     * @author Stian Soiland-Reyes
048     */
049    public class PropertyLiteral implements PropertyObject {

135            protected static DocumentBuilder getDocumentBuilder() {
136                    if (docBuilder == null) {
137                            synchronized (PropertyLiteral.class) {
138                                    if (docBuilder != null) {
139                                            return docBuilder;
140                                    }
141                                    DocumentBuilderFactory factory = getDocumentBuilderFactory();
142                                    factory.setNamespaceAware(true);
143                                    factory.setValidating(false);
144                                    try {
145                                            docBuilder = factory.newDocumentBuilder();
146                                    } catch (ParserConfigurationException e) {
147                                            throw new IllegalStateException("Can't create DocumentBuilder", e);
148                                    }
149                            }
150                    }
151                    return docBuilder;
152            }
153    
154            protected static Transformer getTransformer() {
155                    if (transformer == null) {
156                            synchronized (PropertyLiteral.class) {
157                                    if (transformer != null) {
158                                            return transformer;
159                                    }
160                                    TransformerFactory tFactory = getTransformerFactory();
161                                    try {
162                                            transformer = tFactory.newTransformer();
163                                    } catch (TransformerConfigurationException e1) {
164                                            throw new IllegalStateException("Can't find transformer to write XML", e1);
165                                    }
166                            }
167                    }
168                    return transformer;
169            }

223                    // DocumentBuilderFactory factory =
224                    // DocumentBuilderFactory.newInstance();

235                    Transformer transformer = getTransformer();
236                    CharArrayWriter writer = new CharArrayWriter();
237                    try {
238                            transformer.transform(new DOMSource(element), new StreamResult(writer));
239                    } catch (TransformerException e) {
240                            throw new IllegalStateException("Can't write XML", e);
241                    }
242                    setLiteralValue(writer.toString());
243            }

349            /**
350             * Returns the literal value as an XML <code>Element</code>.
351             * 
352             * @return the literal value as an XML <code>Element</code>
353             */
354            public Element getLiteralValueAsElement() {
355                    DocumentBuilder docBuilder = getDocumentBuilder();
356                    StringReader reader = new StringReader(getLiteralValue());
357                    Document valueDoc;
358                    try {
359                            valueDoc = docBuilder.parse(new InputSource(reader));
360                    } catch (SAXException e) {
361                            throw new IllegalStateException("Can't parse literal XML:\n" + getLiteralValue(), e);
362                    } catch (IOException e) {
363                            throw new IllegalStateException("Can't parse literal XML:\n" + getLiteralValue(), e);
364                    }
365                    return valueDoc.getDocumentElement();
366            }
