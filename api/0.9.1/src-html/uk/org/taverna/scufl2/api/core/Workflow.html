001    package uk.org.taverna.scufl2.api.core;
002    
003    import java.net.URI;
004    import java.util.ArrayList;
005    import java.util.Collection;
006    import java.util.Iterator;
007    import java.util.List;
008    import java.util.Set;
009    import java.util.TreeSet;
010    import java.util.UUID;
011    
012    import uk.org.taverna.scufl2.api.common.AbstractNamedChild;
013    import uk.org.taverna.scufl2.api.common.Child;
014    import uk.org.taverna.scufl2.api.common.NamedSet;
015    import uk.org.taverna.scufl2.api.common.Ported;
016    import uk.org.taverna.scufl2.api.common.Visitor;
017    import uk.org.taverna.scufl2.api.common.WorkflowBean;
018    import uk.org.taverna.scufl2.api.container.WorkflowBundle;
019    import uk.org.taverna.scufl2.api.port.InputWorkflowPort;
020    import uk.org.taverna.scufl2.api.port.OutputWorkflowPort;
021    
022    /**
023     * A <code>Workflow</code> is a set of {@link Processor}s and {@link DataLink}s between the
024     * <code>Processor</code>s. <code>Workflow</code>s may also have input and output ports.
025     * 
026     * @author Alan R Williams
027     */
028    public class Workflow extends AbstractNamedChild implements Child<WorkflowBundle>, Ported {
029    
030            public static final URI WORKFLOW_ROOT = URI.create("http://ns.taverna.org.uk/2010/workflow/");
031    
032            public static URI generateIdentifier() {
033                    return WORKFLOW_ROOT.resolve(UUID.randomUUID().toString() + "/");
034            }
035    
036            private final TreeSet<DataLink> dataLinks = new TreeSet<DataLink>();
037    
038            private final TreeSet<ControlLink> controlLinks = new TreeSet<ControlLink>();
039    
040            private final NamedSet<InputWorkflowPort> inputPorts = new NamedSet<InputWorkflowPort>();
041            private final NamedSet<OutputWorkflowPort> outputPorts = new NamedSet<OutputWorkflowPort>();
042            private final NamedSet<Processor> processors = new NamedSet<Processor>();
043            private URI workflowIdentifier;
044            private WorkflowBundle parent;
045    
046            /**
047             * Constructs a <code>Workflow</code> with a name based on a random UUID.
048             */
049            public Workflow() {
050                    setWorkflowIdentifier(generateIdentifier());
051                    String workflowId = WORKFLOW_ROOT.relativize(getWorkflowIdentifier()).toASCIIString();
052                    setName("wf-" + workflowId);
053            }
054    
055            @Override
056            public boolean accept(Visitor visitor) {
057                    if (visitor.visitEnter(this)) {
058                            List<Iterable<? extends WorkflowBean>> children = new ArrayList<Iterable<? extends WorkflowBean>>();
059                            children.add(getInputPorts());
060                            children.add(getOutputPorts());
061                            children.add(getProcessors());
062                            children.add(getDataLinks());
063                            children.add(getControlLinks());
064                            outer: for (Iterable<? extends WorkflowBean> it : children) {
065                                    for (WorkflowBean bean : it) {
066                                            if (!bean.accept(visitor)) {
067                                                    break outer;
068                                            }
069                                    }
070                            }
071                    }
072                    return visitor.visitLeave(this);
073            }
074    
075            /**
076             * Returns the <code>ControlLink</code>s.
077             * 
078             * If there are no <code>ControlLink</code>s an empty set is returned.
079             * 
080             * @return the <code>ControlLink</code>s
081             */
082            public Set<ControlLink> getControlLinks() {
083                    return controlLinks;
084            }
085    
086            /**
087             * Returns the <code>DataLink</code>s.
088             * 
089             * If there are no <code>DataLink</code>s an empty set is returned.
090             * 
091             * @return the <code>DataLink</code>s.
092             */
093            public Set<DataLink> getDataLinks() {
094                    return dataLinks;
095            }
096    
097            /**
098             * Returns the <code>InputWorkflowPort</code>s.
099             * 
100             * If there are no <code>InputWorkflowPort</code>s an empty set is returned.
101             * 
102             * @return the <code>InputWorkflowPort</code>s.
103             */
104            @Override
105            public NamedSet<InputWorkflowPort> getInputPorts() {
106                    return inputPorts;
107            }
108    
109            /**
110             * Returns the <code>OutputWorkflowPort</code>s.
111             * 
112             * If there are no <code>OutputWorkflowPort</code>s an empty set is returned.
113             * 
114             * @return the <code>OutputWorkflowPort</code>s.
115             */
116            @Override
117            public NamedSet<OutputWorkflowPort> getOutputPorts() {
118                    return outputPorts;
119            }
120    
121            @Override
122            public WorkflowBundle getParent() {
123                    return parent;
124            }
125    
126            /**
127             * Returns the <code>Processor</code>s.
128             * 
129             * If there are no <code>Processor</code>s an empty set is returned.
130             * 
131             * @return the <code>Processor</code>s.
132             */
133            public NamedSet<Processor> getProcessors() {
134                    return processors;
135            }
136    
137            /**
138             * Returns the workflow identifier.
139             * <p>
140             * The the default identifier is {@value #WORKFLOW_ROOT} plus a random UUID.
141             * @see {@link #setWorkflowIdentifier(URI)}
142             * 
143             * @return the workflow identifier
144             */
145            public URI getWorkflowIdentifier() {
146                    return workflowIdentifier;
147            }
148    
149            /**
150             * Set the <code>ControlLink</code>s to be the contents of the specified set.
151             * <p>
152             * <code>ControlLink</code>s can be added by using {@link #getControlLinks()}.add(controlLink).
153             * 
154             * @param controlLinks the <code>ControlLink</code>s. <strong>Must not</strong> be null
155             */
156            public void setControlLinks(Set<ControlLink> controlLinks) {
157                    this.controlLinks.clear();
158                    this.controlLinks.addAll(controlLinks);
159            }
160    
161            /**
162             * Set the <code>DataLink</code>s to be the contents of the specified set.
163             * <p>
164             * <code>DataLink</code>s can be added by using {@link #getDataLinks()}.add(dataLink).
165             * 
166             * @param dataLinks the <code>DataLink</code>s. <strong>Must not</strong> be null
167             */
168            public void setDataLinks(Set<DataLink> dataLinks) {
169                    dataLinks.clear();
170                    dataLinks.addAll(dataLinks);
171            }
172    
173            /**
174             * Set the <code>InputWorkflowPort</code>s to be the contents of the specified set.
175             * <p>
176             * <code>InputWorkflowPort</code>s can be added by using {@link #getInputWorkflowPorts()}.add(inputPort).
177             * 
178             * @param inputPorts the <code>InputWorkflowPort</code>s. <strong>Must not</strong> be null
179             */
180            public void setInputPorts(Set<InputWorkflowPort> inputPorts) {
181                    this.inputPorts.clear();
182                    for (InputWorkflowPort inputPort : inputPorts) {
183                            inputPort.setParent(this);
184                    }
185            }
186    
187            /**
188             * Set the <code>OutputWorkflowPort</code>s to be the contents of the specified set.
189             * <p>
190             * <code>OutputWorkflowPort</code>s can be added by using {@link #getOutputWorkflowPorts()}.add(outputPort).
191             * 
192             * @param outputPorts the <code>OutputWorkflowPort</code>s. <strong>Must not</strong> be null
193             */
194            public void setOutputPorts(Set<OutputWorkflowPort> outputPorts) {
195                    this.outputPorts.clear();
196                    for (OutputWorkflowPort outputPort : outputPorts) {
197                            outputPort.setParent(this);
198                    }
199            }
200    
201            @Override
202            public void setParent(WorkflowBundle parent) {
203                    if (this.parent != null && this.parent != parent) {
204                            this.parent.getWorkflows().remove(this);
205                    }
206                    this.parent = parent;
207                    if (parent != null) {
208                            parent.getWorkflows().add(this);
209                    }
210    
211            }
212    
213            /**
214             * Set the <code>Processor</code>s to be the contents of the specified set.
215             * <p>
216             * <code>Processor</code>s can be added by using {@link #getProcessors()}.add(processor).
217             * 
218             * @param processors the <code>Processor</code>s. <strong>Must not</strong> be null
219             */
220            public void setProcessors(Set<Processor> processors) {
221                    this.processors.clear();
222                    for (Processor processor : processors) {
223                            processor.setParent(this);
224                    }
225            }
226    
227            /**
228             * Sets the workflow identifier.
229             * 
230             * @param workflowIdentifier the workflow identifier
231             */
232            public void setWorkflowIdentifier(URI workflowIdentifier) {
233                    this.workflowIdentifier = workflowIdentifier;
234            }
235    
236            @Override
237            public String toString() {
238                    final int maxLen = 6;
239                    return "Workflow [getName()=" + getName() + ", getDatalinks()="
240                    + (getDataLinks() != null ? toString(getDataLinks(), maxLen) : null)
241                    + ", getInputPorts()="
242                    + (getInputPorts() != null ? toString(getInputPorts(), maxLen) : null)
243                    + ", getOutputPorts()="
244                    + (getOutputPorts() != null ? toString(getOutputPorts(), maxLen) : null)
245                    + ", getProcessors()="
246                    + (getProcessors() != null ? toString(getProcessors(), maxLen) : null) + "]";
247            }
248    
249            private String toString(Collection<?> collection, int maxLen) {
250                    StringBuilder builder = new StringBuilder();
251                    builder.append("[");
252                    int i = 0;
253                    for (Iterator<?> iterator = collection.iterator(); iterator.hasNext() && i < maxLen; i++) {
254                            if (i > 0) {
255                                    builder.append(", ");
256                            }
257                            builder.append(iterator.next());
258                    }
259                    builder.append("]");
260                    return builder.toString();
261            }
