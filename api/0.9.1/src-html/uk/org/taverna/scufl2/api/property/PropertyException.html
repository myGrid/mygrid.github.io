001    package uk.org.taverna.scufl2.api.property;
002    
003    import java.net.URI;
004    
005    /**
006     * Thrown when an error occurs while trying to retrieve a property.
007     * 
008     * @see PropertyNotFoundException
009     * @see MultiplePropertiesException
010     * @see UnexpectedPropertyException
011     * 
012     * @author Stian Soiland-Reyes
013     */
014    public abstract class PropertyException extends Exception {
015    
016            private static final long serialVersionUID = -3331261486445335308L;
017    
018            private final URI predicate;
019            private final PropertyResource propertyResource;
020    
021            public PropertyException(String message, URI predicate,
022                            PropertyResource propertyResource) {
023                    super(message);
024                    this.predicate = predicate;
025                    this.propertyResource = propertyResource;
026            }
027    
028            public URI getPredicate() {
029                    return predicate;
030            }
031    
032            public PropertyResource getPropertyResource() {
033                    return propertyResource;
034            }
035    
036    }
