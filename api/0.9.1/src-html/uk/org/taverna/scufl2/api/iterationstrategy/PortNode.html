001    package uk.org.taverna.scufl2.api.iterationstrategy;
002    
003    import java.util.List;
004    
005    import uk.org.taverna.scufl2.api.common.Visitor;
006    import uk.org.taverna.scufl2.api.port.InputProcessorPort;
007    
008    public class PortNode implements IterationStrategyNode {
009            private InputProcessorPort inputProcessorPort;
010    
011            private IterationStrategyParent parent;
012            private Integer desiredDepth;
013    
014            public PortNode() {
015            }
016    
017            public PortNode(IterationStrategyParent parent,
018                            InputProcessorPort inputProcessorPort) {
019                    setParent(parent);
020                    setInputProcessorPort(inputProcessorPort);
021            }
022    
023            @Override
024            public boolean accept(Visitor visitor) {
025                    return visitor.visit(this);
026            }
027    
028            public Integer getDesiredDepth() {
029                    return desiredDepth;
030            }
031    
032            public InputProcessorPort getInputProcessorPort() {
033                    return inputProcessorPort;
034            }
035    
036            @Override
037            public IterationStrategyParent getParent() {
038                    return parent;
039            }
040    
041            public void setDesiredDepth(Integer desiredDepth) {
042                    this.desiredDepth = desiredDepth;
043            }
044    
045            public void setInputProcessorPort(InputProcessorPort inputProcessorPort) {
046                    this.inputProcessorPort = inputProcessorPort;
047            }
048    
049            @Override
050            public void setParent(IterationStrategyParent newParent) {
051                    if (parent == newParent) {
052                            return;
053                    }
054    
055                    if (parent != null) {
056                            // Remove from old parent
057                            if (!(parent instanceof DotProduct)
058                                            && !(parent instanceof CrossProduct)) {
059                                    throw new IllegalArgumentException(
060                                                    "Old PortNode parent must be a DotProduct or CrossProduct: "
061                                                    + parent);
062                            }
063                            @SuppressWarnings("unchecked")
064                            List<IterationStrategyNode> parentList = (List<IterationStrategyNode>) parent;
065                            parentList.remove(this);
066                    }
067    
068                    parent = newParent;
069    
070                    if (!(parent instanceof DotProduct)
071                                    && !(parent instanceof CrossProduct)) {
072                            throw new IllegalArgumentException(
073                                            "PortNode parent must be a DotProduct or CrossProduct: "
074                                            + parent);
075                    }
076                    @SuppressWarnings("unchecked")
077                    List<IterationStrategyNode> parentList = (List<IterationStrategyNode>) parent;
078                    if (!parentList.contains(this)) {
079                            parentList.add(this);
080                    }
081    
082            }
083    }
