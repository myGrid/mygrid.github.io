001    package uk.org.taverna.scufl2.api.io;
002    
003    import java.io.File;
004    import java.io.IOException;
005    import java.io.OutputStream;
006    import java.util.Set;
007    
008    import uk.org.taverna.scufl2.api.container.WorkflowBundle;
009    
010    /**
011     * A writer for {@link WorkflowBundle}s.
012     * 
013     * Implementations specify workflow bundle formats (media types) that they can write.
014     */
015    public interface WorkflowBundleWriter {
016    
017            /**
018             * Returns the media types that this writer can handle.
019             * 
020             * @return the media types that this writer can handle
021             */
022            public Set<String> getMediaTypes();
023    
024            /**
025             * Writes a <code>WorkflowBundle</code> to a file with specified media type.
026             * 
027             * @param wfBundle
028             *            the workflow bundle to write
029             * @param destination
030             *            the file to write the workflow bundle to
031             * @param mediaType
032             *            the media type to write workflow bundle in
033             * @throws WriterException
034             *             if there is an error writing the workflow bundle
035             * @throws IOException
036             *             if there is an error writing the file
037             */
038            public void writeBundle(WorkflowBundle wfBundle, File destination, String mediaType)
039            throws WriterException, IOException;
040    
041            /**
042             * Writes a <code>WorkflowBundle</code> to a stream with specified media type.
043             * 
044             * @param wfBundle
045             *            the workflow bundle to write
046             * @param output
047             *            the stream to write the workflow bundle to
048             * @param mediaType
049             *            the media type to write workflow bundle in
050             * @throws WriterException
051             *             if there is an error writing the workflow bundle
052             * @throws IOException
053             *             if there is an error writing to the stream
054             */
055            public void writeBundle(WorkflowBundle wfBundle, OutputStream output, String mediaType)
056            throws WriterException, IOException;
057    
058    }
