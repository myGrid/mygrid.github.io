021    package uk.org.taverna.scufl2.api.configurations;
022    
023    import java.net.URI;
024    
025    import uk.org.taverna.scufl2.api.property.PropertyResource;
026    
027    /**
028     * Definition of the {@link Configuration} required to configure a
029     * {@link uk.org.taverna.scufl2.api.common.Configurable Configurable}.
030     * 
031     * @author David Withers
032     */
033    public class ConfigurationDefinition {
034    
035            private URI configurableType;
036    
037            private final PropertyResourceDefinition propertyResourceDefinition;
038    
039            /**
040             * Creates the definition of a {@link Configuration}.
041             */
042            public ConfigurationDefinition() {
043                    propertyResourceDefinition = new PropertyResourceDefinition();
044            }
045    
046            /**
047             * Creates the definition of the {@link Configuration} required to configure the
048             * {@link uk.org.taverna.scufl2.api.common.Configurable Configurable} specified by the
049             * configurableType.
050             * 
051             * @param configurableType
052             *            the URI that identifies the <code>Configurable</code> type
053             */
054            public ConfigurationDefinition(URI configurableType) {
055                    this.configurableType = configurableType;
056                    propertyResourceDefinition = new PropertyResourceDefinition();
057                    propertyResourceDefinition.setTypeURI(configurableType.resolve("#Config"));
058                    propertyResourceDefinition.setDescription("");
059                    propertyResourceDefinition.setRequired(true);
060                    propertyResourceDefinition.setMultiple(false);
061                    propertyResourceDefinition.setOrdered(false);
062            }
063    
064            /**
065             * Returns the URI that identifies the {@link uk.org.taverna.scufl2.api.common.Configurable
066             * Configurable} type.
067             * 
068             * @return the URI that identifies the <code>Configurable</code> type
069             */
070            public URI getConfigurableType() {
071                    return configurableType;
072            }
073    
074            /**
075             * Returns the definition of the {@link PropertyResource} required by the
076             * <code>Configuration</code>.
077             * 
078             * @return the definition of the {@link PropertyResource} required by the
079             *         <code>Configuration</code>
080             */
081            public PropertyResourceDefinition getPropertyResourceDefinition() {
082                    return propertyResourceDefinition;
083            }
084    
085            /**
086             * Sets the URI that identifies the {@link uk.org.taverna.scufl2.api.common.Configurable
087             * Configurable} type.
088             * 
089             * @param configurableType
090             *            the URI that identifies the <code>Configurable</code> type
091             */
092            public void setConfigurableType(URI configurableType) {
093                    this.configurableType = configurableType;
094            }
095    
096            @Override
097            public String toString() {
098                    StringBuilder sb = new StringBuilder();
099                    sb.append("ConfigurationDefinition for ");
100                    sb.append(configurableType);
101                    sb.append("\n");
102                    sb.append(propertyResourceDefinition);
103                    sb.append("\n");
104                    return sb.toString();
105            }
106    
107    }
