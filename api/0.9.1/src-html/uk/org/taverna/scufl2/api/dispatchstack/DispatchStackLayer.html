001    package uk.org.taverna.scufl2.api.dispatchstack;
002    
003    import java.net.URI;
004    
005    import uk.org.taverna.scufl2.api.common.Child;
006    import uk.org.taverna.scufl2.api.common.Configurable;
007    import uk.org.taverna.scufl2.api.common.Typed;
008    import uk.org.taverna.scufl2.api.common.Visitor;
009    
010    /**
011     * A <code>DispatchStackLayer</code> adds functionality to a
012     * <code>DispatchStack</code>.
013     * <p>
014     * e.g. a retry layer (of type
015     * http://ns.taverna.org.uk/2010/scufl2/taverna#Failover) adds the functionality
016     * to retry an Activity if an invocation fails.
017     */
018    public class DispatchStackLayer implements Typed, Child<DispatchStack>,
019                    Configurable {
020    
021            private DispatchStack parent;
022            private URI configurableType;
023    
024            /**
025             * Constructs a <code>DispatchStackLayer</code>.
026             */
027            public DispatchStackLayer() {
028            }
029    
030            /**
031             * Constructs a <code>DispatchStackLayer</code> for the specified
032             * <code>DispatchStack</code> with the specified type.
033             * 
034             * @param parent
035             *            the <code>DispatchStack</code> to add the
036             *            <code>DispatchStackLayer</code> to
037             * @param configurableType
038             *            the type of the <code>DispatchStackLayer</code>
039             */
040            public DispatchStackLayer(DispatchStack parent, URI configurableType) {
041                    setParent(parent);
042                    setConfigurableType(configurableType);
043            }
044    
045            @Override
046            public boolean accept(Visitor visitor) {
047                    return visitor.visit(this);
048            }
049    
050            /**
051             * Returns the type of the <code>DispatchStackLayer</code>.
052             * 
053             * @return the type of the <code>DispatchStackLayer</code>
054             */
055            @Override
056            public URI getConfigurableType() {
057                    return configurableType;
058            }
059    
060            @Override
061            public DispatchStack getParent() {
062                    return parent;
063            }
064    
065            /**
066             * Sets the type of the <code>DispatchStackLayer</code>.
067             * 
068             * @param type
069             *            the type of the <code>DispatchStackLayer</code>
070             */
071            @Override
072            public void setConfigurableType(URI type) {
073                    configurableType = type;
074            }
075    
076            @Override
077            public void setParent(DispatchStack parent) {
078                    if (this.parent == parent) {
079                            return; // No more to do!
080                    }
081                    if (this.parent != null) {
082                            this.parent.remove(this);
083                    }
084                    this.parent = parent;
085                    if (parent != null && !parent.contains(this)) {
086                            parent.add(this); // Just add to the end
087                    }
088            }
089    
090            @Override
091            public String toString() {
092                    StringBuffer sb = new StringBuffer("DispatchStackLayer");
093                    if (getConfigurableType() != null) {
094                            sb.append(" ");
095                            sb.append(getConfigurableType());
096                    }
097                    if (getParent() != null && getParent().getParent() != null) {
098                            int index = getParent().indexOf(this);
099                            sb.append(" #");
100                            sb.append(index);
101                            sb.append(" in ");
102                            sb.append(getParent().getParent());
103                    }
104                    return sb.toString();
105            }
106            
107    }
